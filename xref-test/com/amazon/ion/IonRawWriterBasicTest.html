41   * Tests the raw string writing APIs and basic behavior of the raw symbol ID
42   * writing APIs provided by the {@link _Private_IonRawWriter} and {@link IonBinaryWriter}

49      private static final IonSystem system = IonSystemBuilder.standard().build();
50  
51      private static IonWriter writer(OutputStream out, SymbolTable...imports)
52      {
53          // Only the binary writer supports the managed/raw facets.
54          return system.newBinaryWriter(out, imports);
55      }
56  
57      static List<String> getStrings(String file)
58      {
59          List<String> strings = new ArrayList<String>();
60          IonReader reader;
61          try
62          {
63              reader = IonReaderBuilder.standard().build(new FileInputStream(IonTestCase.getTestdataFile(file)));
64          }
65          catch (FileNotFoundException e)
66          {
67              // convert to unchecked exception so this plays well in static initializers
68              throw new RuntimeException(e);
69          }
70          while (reader.next() != null)
71          {
72              try
73              {
74                  strings.add(reader.stringValue());
75              }
76              catch (UnknownSymbolException e)
77              {
78                  // If the input file contains unknown symbol text, e.g. $123,
79                  // just skip it.
80              }
81          }
82          return strings;
83      }
84  
85      static abstract class Roundtrip {
86  
87          private final SymbolTable[] imports;
88          private final SimpleCatalog catalog;
89  
90          public Roundtrip(SymbolTable...imports)
91          {
92              this.imports = imports;
93              catalog = new SimpleCatalog();
94              for (SymbolTable imported : imports)
95              {
96                  catalog.putTable(imported);
97              }
98          }
99  
100         public void test() throws IOException
101         {
102             ByteArrayOutputStream out = new ByteArrayOutputStream();
103             IonWriter writer = writer(out, imports);
104             _Private_IonManagedWriter managedWriter = writer.asFacet(_Private_IonManagedWriter.class);
105             _Private_IonRawWriter rawWriter = managedWriter.getRawWriter();
106             write(managedWriter, rawWriter);
107             writer.close();
108             read(IonReaderBuilder.standard().withCatalog(catalog).build(out.toByteArray()));
109         }
110 
111         abstract void write(_Private_IonManagedWriter managedWriter, _Private_IonRawWriter rawWriter) throws IOException;
112         abstract void read(IonReader reader);
113     }

115     private void testWriteEncodedString(final String str, final int padding) throws IOException
116     {
117         new Roundtrip()
118         {
119             int written = 0;
120 
121             @Override
122             void write(_Private_IonManagedWriter managedWriter, _Private_IonRawWriter rawWriter)
123                 throws IOException
124             {
125                 byte[] encoded = str.getBytes("UTF-8");
126                 int encodedLength = encoded.length;
127 
128                 rawWriter.writeString(encoded, 0, encodedLength);
129                 written++;
130 
131                 byte[] block = new byte[encodedLength + padding];
132                 for (int i = 0; i < 3; i++)
133                 {
134                     // Write from buffer with 'padding' extra bytes. 'shift' indicates
135                     // the position at which the actual data starts.
136                     int shift = padding * i / 2;
137                     System.arraycopy(encoded, 0, block, shift, encodedLength);
138                     rawWriter.writeString(block, shift, encodedLength);
139                     written++;
140                 }
141             }
142 
143             @Override
144             void read(IonReader reader)
145             {
146                 int read = 0;
147                 while (reader.next() != null)
148                 {
149                     assertEquals(IonType.STRING, reader.getType());
150                     assertEquals(str, reader.stringValue());
151                     read++;
152                 }
153                 assertEquals(written, read);
154             }
155 
156         }.test();
157     }

159     @Test
160     public void testWriteEncodedStrings() throws IOException
161     {
162         int i = 0;
163         for (String str : getStrings("good/strings.ion"))
164         {
165             testWriteEncodedString(str, i * 2);
166             i++;
167         }

172     {

175

180                 rawWriter.writeString(null, 1, 2); // position arguments ignored
181             }
182 
183             @Override
184             void read(IonReader reader)
185             {
186                 assertEquals(IonType.STRING, reader.next());
187                 assertTrue(reader.isNullValue());
188                 assertNull(reader.stringValue());
189             }

194     private _Private_IonRawWriter basicRawWriter()

204     @Test
205     public void testRawWriteIVMFails() throws IOException
206     {
207         _Private_IonRawWriter rawWriter = basicRawWriter();
208         rawWriter.addTypeAnnotationSymbol(10);
209         rawWriter.writeSymbolToken(2); // Symbol 2 with an annotation is not IVM
210         rawWriter.stepIn(IonType.STRUCT);
211         rawWriter.setFieldNameSymbol(11);
212         rawWriter.writeSymbolToken(2); // Symbol 2 below top level is not IVM
213         rawWriter.stepOut();
214         thrown.expect(IonException.class);
215         rawWriter.writeSymbolToken(2); // Top-level symbol 2 is IVM
216     }

226     @Test
227     public void testRawSetTypeAnnotationSymbolsOverwrites() throws IOException
228     {
229         new Roundtrip()
230         {
231 
232             @Override
233             void write(_Private_IonManagedWriter managedWriter, _Private_IonRawWriter rawWriter)
234                 throws IOException
235             {
236                 managedWriter.requireLocalSymbolTable();
237                 rawWriter.addTypeAnnotationSymbol(4);
238                 rawWriter.addTypeAnnotationSymbol(5);
239                 rawWriter.setTypeAnnotationSymbols(6, 7);
240                 rawWriter.writeInt(42); // dummy value
241             }
242 
243             @Override
244             void read(IonReader reader)
245             {
246                 reader.next();
247                 SymbolToken[] annotations = reader.getTypeAnnotationSymbols();
248                 assertEquals(2, annotations.length);
249                 int[] annotationSids = { annotations[0].getSid(), annotations[1].getSid() };
250                 assertArrayEquals(new int[]{6, 7}, annotationSids);
251             }
252 
253         }.test();
