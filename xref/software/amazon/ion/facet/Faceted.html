15  package software.amazon.ion.facet;
16  
17  /**
18   * Provides access to optional extension interfaces of a subject instance.
19   * Users can request a facet of the subject by passing the desired type token
20   * to {@link #asFacet(Class)}. Different implementations, or even different
21   * instances, of a subject may support different facets.
22   * Consult the subject's documentation to determine which facets are available
23   * in each circumstance.
24   *
25   * <h2>Design Notes</h2>
26   *
27   * Given a concrete {@link Faceted} class, it may be that some instances
28   * support a particular facet while others do not, depending on the state of
29   * the subject or the way it was constructed. In such cases
30   * {@link #asFacet(Class) asFacet} should choose whether to return the facet
31   * based on the subject's state.
32   * Such classes should <em>not</em> extend the facet
33   * interface (directly or indirectly), since that allows clients to bypass
34   * {@link #asFacet(Class) asFacet} and simply downcast the subject to the facet,
35   * causing problems for instances that can't support the facet.
36   *
37   * @see Facets
38   *
39   */
40  public interface Faceted
41  {
42      /**
43       * Returns a facet of this subject if supported.
44       *
45       * @param <T>       The requested facet type.
46       * @param facetType The type token of the requested facet type.
47       *
48       * @return  An instance of T representing the facet of the subject, or
49       *          null if the facet is not supported by the subject.
50       *
51       * @see Facets#asFacet(Class, Faceted)
52       * @see Facets#assumeFacet(Class, Faceted)
53       */
54      public <T> T asFacet(Class<T> facetType);
55  }
