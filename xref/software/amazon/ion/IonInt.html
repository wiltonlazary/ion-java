15  package software.amazon.ion;
16  
17  import java.math.BigInteger;
18  
19  /**
20   * An Ion <code>int</code> value.
21   * <p>
22   * <b>WARNING:</b> This interface should not be implemented or extended by
23   * code outside of this library.
24   */
25  public interface IonInt
26      extends IonValue
27  {
28      /**
29       * Gets the content of this Ion <code>int</code> as a Java
30       * <code>int</code> value.
31       *
32       * @return the int value.
33       * @throws NullValueException if <code>this.isNullValue()</code>.
34       */
35      public int intValue()
36          throws NullValueException;
37  
38      /**
39       * Gets the content of this Ion <code>int</code> as a Java
40       * <code>long</code> value.
41       *
42       * @return the long value.
43       * @throws NullValueException if <code>this.isNullValue()</code>.
44       */
45      public long longValue()
46          throws NullValueException;
47  
48  
49      /**
50       * Gets the content of this Ion <code>int</code> as a Java
51       * {@link BigInteger} value.
52       *
53       * @return the <code>BigInteger</code> value,
54       * or <code>null</code> if this is <code>null.int</code>.
55       */
56      public BigInteger bigIntegerValue();
57  
58      /**
59       * Gets an {@link IntegerSize} representing the smallest-possible
60       * Java type of the underlying content, or {@code null} if this is
61       * {@code null.int}.
62       *
63       * @see IonReader#getIntegerSize()
64       */
65      public IntegerSize getIntegerSize();
66  
67      /**
68       * Sets the content of this value.
69       */
70      public void setValue(int value);
71  
72      /**
73       * Sets the content of this value.
74       */
75      public void setValue(long value);
76  
77      /**
78       * Sets the content of this value.
79       * The integer portion of the number is used, any fractional portion is
80       * ignored.
81       *
82       * @param content the new content of this int;
83       * may be <code>null</code> to make this <code>null.int</code>.
84       */
85      public void setValue(Number content);
86  
87      public IonInt clone()
88          throws UnknownSymbolException;
89  }
