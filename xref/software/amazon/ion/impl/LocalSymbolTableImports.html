17  import java.util.ArrayList;
18  import static software.amazon.ion.SymbolTable.UNKNOWN_SYMBOL_ID;
19  
20  import java.util.Arrays;
21  import java.util.List;
22  import software.amazon.ion.SymbolTable;
23  import software.amazon.ion.SymbolToken;

25  /**
26   * This class manages the system symbol table and any shared symbol table(s)
27   * imported by a local symbol table. It provides "find" methods to find
28   * either symbol Ids or names in the imported tables.
29   * <p>
30   * This class is <b>immutable</b>, and hence safe for use by multiple threads.
31   */
32  // TODO amzn/ion-java#37 Create specialized class to handle the common case where
33  //      there are zero or one imported non-system shared symtab(s).
34  final class LocalSymbolTableImports
35  {
36      /**
37       * The symtabs imported by a local symtab, never null or empty. The first
38       * symtab must be a system symtab, the rest must be non-system shared
39       * symtabs.
40       */
41      private final SymbolTable[] myImports;
42  
43      /**
44       * The maxId of all imported tables, i.e., the sum of all maxIds declared
45       * by symtabs in {@link #myImports}.
46       */
47      private final int           myMaxId;
48  
49      /**
50       * The base Sid of each symtab in {@link #myImports} in parallel, i.e.,
51       * {@link #myBaseSids}[0] references {@link #myImports}[0]. Must be
52       * the same length as {@link #myImports}.
53       */
54      private final int[]         myBaseSids;
55  
56      //==========================================================================
57      // Constructor(s) and static factory methods
58      //==========================================================================
59  
60      /**
61       * Constructor, takes the passed-in {@code importTables} containing the
62       * imported symtabs.
63       *
64       * @param importTables
65       *          the imported symtabs, must contain at least one element; the
66       *          first element must be a system symtab, the rest must be
67       *          non-system shared symtabs
68       *
69       * @throws IllegalArgumentException
70       *          if any import is a local table, or if any but the first is a
71       *          system table
72       * @throws NullPointerException
73       *          if any import is null
74       */
75      LocalSymbolTableImports(List<SymbolTable> importTables)
76      {
77          validateImports(importTables);
78  
79          int importTablesSize = importTables.size();
80  
81          // detects and adapts local tables so they are importable
82          myImports = new SymbolTable[importTablesSize];
83          for(int i = 0; i < importTables.size(); i++)
84          {
85              SymbolTable symbolTable = importTables.get(i);
86              if(symbolTable.isLocalTable())
87              {
88                  myImports[i] = LocalSymbolTableImportAdapter.of((LocalSymbolTable) symbolTable);
89              }
90              else
91              {
92                  myImports[i] = symbolTable;
93              }
94          }
95  
96          myBaseSids = new int[importTablesSize];
97          myMaxId = prepBaseSids(myBaseSids, myImports);
98      }
99  
100     /**
101      * Validates the import list to ensure that if there is a {@link LocalSymbolTable} in it then it's a single import
102      * apart from the system table
103      */
104     private void validateImports(final List<SymbolTable> importTables)
105     {
106         int sizeWithoutSystemTables = importTables.size();
107         int numberOfLocalTables = 0;
108 
109         for(SymbolTable table : importTables)
110         {
111             if(table.isLocalTable()) numberOfLocalTables++;
112             if(table.isSystemTable()) sizeWithoutSystemTables--;
113         }
114 
115         if(numberOfLocalTables > 0 && sizeWithoutSystemTables != 1){
116             throw new IllegalArgumentException("when importing LocalSymbolTables it needs to be the only import");
117         }
118     }
119 
120     /**
121      * @param defaultSystemSymtab
122      *          the default system symtab, which will be used if the first
123      *          import in {@code imports} isn't a system symtab, never null
124      * @param imports
125      *          the set of shared symbol tables to import; the first (and only
126      *          the first) may be a system table, in which case the
127      *          {@code defaultSystemSymtab is ignored}
128      *
129      * @throws IllegalArgumentException
130      *          if any import is a local table, or if any but the first is a
131      *          system table
132      * @throws NullPointerException
133      *          if any import is null
134      */
135     LocalSymbolTableImports(SymbolTable defaultSystemSymtab,
136                             SymbolTable... imports)
137     {
138         assert defaultSystemSymtab.isSystemTable()
139             : "defaultSystemSymtab isn't a system symtab";
140 
141         if (imports != null && imports.length > 0)
142         {
143             final int offset;
144             final SymbolTable systemTable;
145             if (imports[0].isSystemTable())
146             {
147                 offset = 0;
148                 systemTable = imports[0];
149             }
150             else
151             {
152                 offset = 1;
153                 systemTable = defaultSystemSymtab;
154             }
155 
156             myImports = new SymbolTable[imports.length + offset];
157             myImports[0] = systemTable;
158 
159             // no need to consider the offset here as it only comes in play when there is a system table
160             validateImports(Arrays.asList(imports));
161 
162             for(int i = 1 - offset; i < imports.length; i++)
163             {
164                 SymbolTable symbolTable = imports[i];
165                 if(symbolTable instanceof LocalSymbolTable)
166                 {
167                     myImports[i + offset] = LocalSymbolTableImportAdapter.of((LocalSymbolTable) symbolTable);
168                 }
169                 else
170                 {
171                     myImports[i + offset] = symbolTable;
172                 }
173             }
174         }
175         else
176         {
177             // use defaultSystemSymtab only
178             myImports = new SymbolTable[] { defaultSystemSymtab };
179         }
180 
181         myBaseSids = new int[myImports.length];
182         myMaxId = prepBaseSids(myBaseSids, myImports);
183     }
184 
185     /**
186      * Collects the necessary maxId info. from the passed-in {@code imports}
187      * and populates the {@code baseSids} array.
188      *
189      * @return the sum of all imports' maxIds
190      *
191      * @throws IllegalArgumentException
192      *          if any symtab beyond the first is a local or system symtab
193      */
194     private static int prepBaseSids(int[] baseSids, SymbolTable[] imports)
195     {
196         SymbolTable firstImport = imports[0];
197 
198         assert firstImport.isSystemTable()
199             : "first symtab must be a system symtab";
200 
201         baseSids[0] = 0;
202         int total = firstImport.getMaxId();
203 
204         for (int i = 1; i < imports.length; i++)
205         {
206             SymbolTable importedTable = imports[i];
207 
208             if (importedTable.isSystemTable())
209             {
210                 String message = "only non-system shared tables can be imported";
211                 throw new IllegalArgumentException(message);
212             }
213 
214             baseSids[i] = total;
215             total += imports[i].getMaxId();
216         }
217 
218         return total;
219     }

223     String findKnownSymbol(int sid)
224     {
225         String name = null;
226 
227         if (sid <= myMaxId)
228         {
229             int i, previousBaseSid = 0;
230             for (i = 1; i < myImports.length; i++)
231             {
232                 int baseSid = myBaseSids[i];
233                 if (sid <= baseSid)
234                 {
235                     break;
236                 }
237                 previousBaseSid = baseSid;
238             }
239 
240             // if we run over myImports.length, the sid is in the last symtab
241             int importScopedSid = sid - previousBaseSid;
242             name = myImports[i-1].findKnownSymbol(importScopedSid);
243         }
244 
245         return name;
246     }
247 
248     int findSymbol(String name)
249     {
250         SymbolToken tok = find(name);
251         return (tok == null ? UNKNOWN_SYMBOL_ID : tok.getSid());
252     }
253 
254     /**
255      * Finds a symbol already interned by an import, returning the lowest
256      * known SID.
257      * <p>
258      * This method will not necessarily return the same instance given the
259      * same input.
260      *
261      * @param text the symbol text to find
262      *
263      * @return
264      *          the interned symbol (with both text and SID), or {@code null}
265      *          if it's not defined by an imported table
266      */
267     SymbolToken find(String text)
268     {
269         for (int i = 0; i < myImports.length; i++)
270         {
271             SymbolTable importedTable = myImports[i];
272             SymbolToken tok = importedTable.find(text);
273 
274             if (tok != null)
275             {
276                 int sid = tok.getSid() + myBaseSids[i];
277                 text = tok.getText(); // Use interned instance
278 
279                 assert text != null;
280 
281                 return new SymbolTokenImpl(text, sid);
282             }
283         }
284         return null;
285     }

287     int getMaxId()
288     {
289         return myMaxId;
290     }

292     /**
293      * Gets the sole system symtab.
294      */
295     SymbolTable getSystemSymbolTable()
296     {
297         assert myImports[0].isSystemTable();
298         return myImports[0];
299     }
300 
301     /**
302      * Gets all non-system shared symtabs (if any).
303      *
304      * @return a newly allocated copy of the imported symtabs
305      */
306     SymbolTable[] getImportedTables()
307     {
308         int count = myImports.length - 1; // we don't include system symtab
309         SymbolTable#SymbolTable">SymbolTable[] imports = new SymbolTable[count];
310         if (count > 0)
311         {
312             // defensive copy
313             System.arraycopy(myImports, 1, imports, 0, count);
314         }
315         return imports;
316     }

318     /**
319      * Returns the {@link #myImports} member field without making a copy.
320      * <p>
321      * <b>Note:</b> Callers must not modify the resulting SymbolTable array!
322      * This will violate the immutability property of this class.
323      *
324      * @return
325      *          the backing array of imported symtabs, as-is; the first element
326      *          is a system symtab, the rest are non-system shared symtabs
327      *
328      * @see #getImportedTables()
329      */
330     SymbolTable[] getImportedTablesNoCopy()
331     {
332         return myImports;
333     }
334 
335     @Override
336     public String toString()
337     {
338         return Arrays.toString(myImports);
339     }

341     /**
342      * Determines whether the passed-in instance has the same sequence of
343      * symbol table imports as this instance. Note that equality of these
344      * imports are checked using their reference, instead of their semantic
345      * state.
346      */
347     boolean equalImports(LocalSymbolTableImports other)
348     {
349         return Arrays.equals(myImports, other.myImports);
350     }
