149         public Type setNumericValue(IonTokenReader tr, String s) {
150             switch (this) {
151             case kwNan:
152                 tr.doubleValue = Double.NaN;
153                 return this;
154             case kwPosInf:
155                 tr.doubleValue = Double.POSITIVE_INFINITY;
156                 return this;
157             case kwNegInf:
158                 tr.doubleValue = Double.NEGATIVE_INFINITY;
159                 return this;
160             case constNegInt:
161             case constPosInt:
162                 if (NumberType.NT_HEX.equals(tr.numberType)) {
163                     tr.intValue = new BigInteger(s, 16);
164                     // In hex case we've discarded the prefix [+-]?0x so
165                     // reconstruct the sign
166                     if (this == constNegInt) tr.intValue = tr.intValue.negate();
167                 }
168                 else {
169                     tr.intValue = new BigInteger(s, 10);
170 
171                     // Make sure that sign aligns with type.
172                     // Note that this allows negative zero.
173                     assert (BigInteger.ZERO.equals(tr.intValue)
174                              ? true
175                              : this == (tr.intValue.signum() < 0 ? constNegInt : constPosInt));
176                 }
177                 return this;
178             case constFloat:
179                 tr.doubleValue = Double.parseDouble(s);
180                 return this;
181             case constDecimal:
182                 // BigDecimal parses using e instead of d or D
183                 String eFormat = s.replace('d', 'e');
184                 if (eFormat == s) {
185                     // No match for 'd' but look for 'D'
186                     eFormat = s.replace('D', 'e');
187                 }
188                 tr.decimalValue = Decimal.valueOf(eFormat);
189                 return this;
190             case constTime:
191                 tr.dateValue = timeinfo.parse(s);
192                 return this;
193             default:
194                 throw new  AssertionError("Unknown op for numeric case: " + this);
195             }
196         }

219         IonTokenReader _tr;
220         int            _sboffset;
221         int            _sbavailable;
222 
223         LocalReader(IonTokenReader tr) {
224             _tr = tr;
225         }
226 
227         @Override
228         public void close() throws IOException {
229             _tr = null;
230             return;
231         }
232 
233         @Override
234         public void reset() {
235             _sboffset = 0;
236             _sbavailable = _tr.value.length();
237         }
238 
239         @Override
240         public int read() throws IOException {
241             int c = -1;
242 
243             if (_sbavailable > 0) {
244                 c = _tr.value.charAt(_sboffset++);
245                 _sbavailable--;
246             }
247             else {
248                 c = _tr.read();
249             }
250             return c;
251         }
252 
253         @Override
254         public int read(char[] dst, int dstoffset, int len) throws IOException {
255             int needed = len;
256 
257             while (needed-- > 0) {
258                 int c = this.read();
259                 if (c < 0) break;
260                 dst[dstoffset++] = (char)c;
261             }
262             return len - needed;
263         }

408     int readIgnoreWhitespace() throws IOException {
409         assert !inQuotedContent;
410 
411         int c;
412 
413         // read through comments - this detects and rejects double slash
414         // and slash star style comments and their contents
415         for (;;) {
416             c = read();
417 
418             if (c == '/') { // possibly a start of a comment
419                 int c2 = read();
420 
421                 if (c2 == '/') {
422                     // we have a // comment, scan for the terminating new line
423                     while (c2 != '\n' && c2 != -1) {
424                         c2 = read();
425                     }
426                     c = c2;
427                 }
428                 else if (c2 == '*') {
429                     // we have a /* */ comment scan for closing */
430                     scancomment:
431                     while (c2 != -1) {
432                         c2 = read();
433 
434                         if (c2 == '*') {
435                             c2 = read();
436 
437                             if (c2 == '/') {
438                                 break scancomment;
439                             }
440                             unread(c2); // in case this was the '*' we're looking for
441                         }
442                     }
443                     c = read();
444                 }
445                 else {
446                     // it wasn't a comment start, throw it back
447                     unread(c2);
448                 }
449             }
450             if (/* inContent ||*/ !isWhitespace(c)) {
451                 break;
452             }
453         }
454         return c;
455     }

470     private Type next(int c, boolean is_in_expression) throws IOException {
471         int c2;
472         t = Type.none;
473 
474         isIncomplete = false;
475         switch (c) {
476         case -1:
477             return (t = Type.eof);
478         case '{':
479             c2 = read();
480             if (c2 == '{') {
481                 return (t = Type.tOpenDoubleCurly);
482             }
483             unread(c2);
484             return (t = Type.tOpenCurly);
485         case '}':
486             return (t = Type.tCloseCurly);
487         case '[':
488             return (t = Type.tOpenSquare);
489         case ']':
490             return (t = Type.tCloseSquare);
491         case '(':
492             return (t = Type.tOpenParen);
493         case ')':
494             return (t = Type.tCloseParen);
495         case ',':
496             return (t = Type.tComma);
497         case '0': case '1': case '2': case '3': case '4':
498         case '5': case '6': case '7': case '8': case '9':
499             return readNumber(c);
500         case '\"':
501             inQuotedContent = true;
502             this.keyword = Type.none; // anything in quotes isn't a keyword
503             return scanString(c, _Private_IonConstants.lnIsVarLen - 1);
504         case '\'':
505             c2 = read();
506             if (c2 == '\'') {
507                 c2 = read();
508                 if (c2 == '\'') {
509                     return scanLongString();
510                 }
511                 this.unread(c2);
512                 c2 = '\''; // restore c2 for the next unread
513             }
514             this.unread(c2);
515             inQuotedContent = true;
516             return scanIdentifier(c);
517         case '+':
518             c2 = read();
519             if (c2 == 'i') {
520                 c2 = read();
521                 if (c2 == 'n') {
522                     c2 = read();
523                     if (c2 == 'f') {
524                         return (t = Type.kwPosInf);
525                     }
526                     this.unread(c2);
527                     c2 = 'n';
528                 }
529                 this.unread(c2);
530                 c2 = 'i';
531             }
532             this.unread(c2);
533             if (is_in_expression) {
534                 return scanOperator(c);
535             }
536             break; // break to error
537         case '-':
538             c2 = read();
539             if (c2 >= '0' && c2 <= '9') {
540                 this.unread(c2);
541                 return readNumber(c);
542             }
543             if (c2 == 'i') {
544                 c2 = read();
545                 if (c2 == 'n') {
546                     c2 = read();
547                     if (c2 == 'f') {
548                         return (t = Type.kwNegInf);
549                     }
550                     this.unread(c2);
551                     c2 = 'n';
552                 }
553                 this.unread(c2);
554                 c2 = 'i';
555             }
556             this.unread(c2);
557             if (is_in_expression) {
558                 return scanOperator(c);
559             }
560             break; // break to error
561         default:
562             if (IonTextUtils.isIdentifierStart(c)) {
563                 return scanIdentifier(c);
564             }
565             if (is_in_expression && isOperatorPart(c)) {
566                 return scanOperator(c);
567             }
568         }
569 
570         String message =
571             "Unexpected character " + printCodePointAsString(c) +
572             " encountered at line " + this.getLineNumber() +
573             " column " + this.getColumn();
574         throw new IonException(message);
575     }

578     public Type scanIdentifier(int c) throws IOException
579     {
580         // reset our local value buffer
581         resetValue();
582 
583         // we don't have an identifier type any longer, just string
584         this.t = Type.constSymbol;
585 
586         // some strings are keywords, most are not
587         // mostly we'll guess that it's not a keyword
588         this.keyword = null;
589 
590         // first we read the identifier content into our value buffer
591         // if the value is quoted, use the escape char loop, otherwise
592         // look for the non-identifier character (and throw it back)
593 
594         if (!readIdentifierContents(c)) { // not quoted (true is quoted)
595             // anything in quotes (even single quotes) is not a keyword
596             // and here we're not in quoted content (that is handled above)
597             // so see if it's also a keyword
598             this.keyword = IonTokenReader.matchKeyword(value, 0, value.length());
599             if (this.keyword != null) {
600                 if (this.keyword == Type.kwNull) {
601                     c = this.read();
602                     if (c == '.') {
603                         int dot = value.length();
604                         value.append((char)c);
605                         c = this.read();
606                         int added = readIdentifierContents(c, IonTokenConstsX.TN_MAX_NAME_LENGTH + 1); // +1 is "enough roap" so if there are extra letters at the end of the keyword we keep at least 1
607                         int kw = IonTokenConstsX.keyword(value, dot+1, dot+added+1);
608                         switch (kw) {
609                         case IonTokenConstsX.KEYWORD_NULL:
610                         case IonTokenConstsX.KEYWORD_BOOL:
611                         case IonTokenConstsX.KEYWORD_INT:
612                         case IonTokenConstsX.KEYWORD_FLOAT:
613                         case IonTokenConstsX.KEYWORD_DECIMAL:
614                         case IonTokenConstsX.KEYWORD_TIMESTAMP:
615                         case IonTokenConstsX.KEYWORD_SYMBOL:
616                         case IonTokenConstsX.KEYWORD_STRING:
617                         case IonTokenConstsX.KEYWORD_BLOB:
618                         case IonTokenConstsX.KEYWORD_CLOB:
619                         case IonTokenConstsX.KEYWORD_LIST:
620                         case IonTokenConstsX.KEYWORD_SEXP:
621                         case IonTokenConstsX.KEYWORD_STRUCT:
622                             this.keyword = setNullType(value, dot + 1, value.length() - dot - 1);
623                             break;
624                         default:
625                             // not a valid type name - so we have to unread back to the dot
626                             for (int ii=value.length(); ii>dot; ) {
627                                 ii--;
628                                 char uc = value.charAt(ii);
629                                 unread(uc);
630                             }
631 
632                             String message =
633                                 position()
634                                 + ": Expected Ion type after 'null.' but found: "
635                                 + value;
636                             throw new IonException(message);
637                         }
638                     }
639                     else {
640                         unread(c);
641                     }
642                 }
643                 this.t = this.keyword;
644                 return this.t;
645             }
646         }
647 
648         // see if we're a user type of a member name
649         c = this.readIgnoreWhitespace();
650         if (c != ':') {
651             unread(c);
652         }
653         else {
654             c = read();
655             if (c != ':') {
656                 unread(c);
657                 this.t = Type.constMemberName;
658             }
659             else {
660                 this.t = Type.constUserTypeDecl;
661             }
662         }
663         return this.t;
664     }

666     boolean readIdentifierContents(int c) throws IOException {
667 
668         int quote = c;
669         inQuotedContent = (quote == '\'' || quote == '\"');
670 
671         if ((quotedIdentifier = inQuotedContent) == true) {
672             for (;;) {
673                 c = read();
674                 if (c < 0 || c == quote) break;
675                 if (c == '\\') {
676                     c = readEscapedCharacter(this.in, false /*not clob*/);
677                 }
678                 if (c != EMPTY_ESCAPE_SEQUENCE) {
679                     value.appendCodePoint(c);
680                 }
681             }
682             if (c == -1) { // TODO throw UnexpectedEofException
683                 throw new IonException("end encountered before closing quote '\\" + (char)endquote+ "'");
684             }
685             // c, at this point is a single quote, which we don't append
686             inQuotedContent = false;
687         }
688         else {
689             value.append((char)c);
690             for (;;) {
691                 c = read();
692                 if (!IonTextUtils.isIdentifierPart(c)) {
693                     break;
694                 }
695                 value.append((char)c);
696             }
697             unread(c);  // we throw back our terminator here
698         }
699         return quotedIdentifier;
700     }
701     int readIdentifierContents(int c, int max_length) throws IOException {
702         assert(c != '\'' && c != '\"');
703 
704         value.append((char)c);

711                 break;
712             }
713             value.append((char)c);
714             count++;
715         }
716 
717         return count;
718     }

945     public Type scanString(int c, int maxlookahead) throws IOException
946     {
947         // reset out local value buffer
948         resetValue();
949 
950         if (c != '\"') {
951             throw new IonException("Programmer error! Only a quote should get you here.");
952         }
953         endquote = '\"';
954 
955         sizedloop:
956         while (maxlookahead-- > 0) {
957             switch ((c = this.read())) {
958             case -1:   break sizedloop; // TODO deoptimize, throw exception
959             case '\"': break sizedloop;
960             case '\n':
961                 throw new IonException("unexpected line terminator encountered in quoted string");
962             case '\\':
963                 c = readEscapedCharacter(this.in, false /*not clob*/);
964                 //    throws UnexpectedEofException on EOF
965                 if (c != EMPTY_ESCAPE_SEQUENCE) {
966                     value.appendCodePoint(c);
967                 }
968                 break;
969             default:
970                 // c could be part of a surrogate pair so we can't use
971                 // appendCodePoint()
972                 value.append((char)c);
973                 break;
974             }
975         }
976 
977         if (maxlookahead != -1 && c == '\"') {
978             // this is the normal, non-longline case so we're just done
979             closeString();
980         }
981         else {
982             // we're not at the closing quote, so this is an incomplete
983             // string which will have to be read to the end ... later
984             leaveOpenString(c, false);
985         }
986         return Type.constString;
987     }

998     /**
999      * Scans the rest of the string through {@link #endquote}.
1000      * <p>
1001        XXX  WARNING  XXX
1002      * Almost identical logic is found in
1003      * {@link IonBinary#appendToLongValue(int, boolean, boolean, PushbackReader)}
1004      *
1005      * @param is_in_expression
1006      * @throws UnexpectedEofException if we hit EOF before the endquote.
1007      */
1008     void finishScanString(boolean is_in_expression) throws IOException
1009     {
1010         assert isIncomplete;
1011         assert inQuotedContent;
1012 
1013         for (;;)
1014         {
1015             int c = this.read();
1016             if (c == -1) {
1017                 throw new UnexpectedEofException();
1018             }
1019             if (c == endquote) break;  // endquote == -1 during long strings
1020             if (c == '\\') {
1021                 c = readEscapedCharacter(this.in, false /*not clob*/);
1022                 if (c != EMPTY_ESCAPE_SEQUENCE) {
1023                     value.appendCodePoint(c);
1024                 }
1025             }
1026             else if (c == '\'' && isLongString) {
1027                 // This happens while handling triple-quote field names
1028                 if (twoMoreSingleQuotes()) {
1029                     // At the end of a long string.  Look for another one.
1030                     inQuotedContent = false;
1031                     c = readIgnoreWhitespace();
1032 
1033                     if (c == '\'' && twoMoreSingleQuotes()) {
1034                         // Yep, another triple-quote.  We've consumed all three
1035                         // so just keep reading.
1036                         inQuotedContent = true;
1037                     }
1038                     else {
1039                         // Something else, we are done.
1040                         unread(c);
1041                         break;
1042                     }
1043                 }
1044                 else {
1045                     // Still inside the long string
1046                     value.append((char)c);
1047                 }
1048             }
1049             else if (!isLongString && (c == '\n')) {
1050                 throw new IonException("unexpected line terminator encountered in quoted string");
1051             }
1052             else {
1053                 value.append((char)c);
1054             }
1055         }
1056         return;
1057     }

1060      * If two single quotes are next on the input, consume them and return
1061      * true.  Otherwise, leave them on the input and return false.
1062      * @return true when there are two pending single quotes.

1110     /**
1111      * @return the translated escape sequence.  Will not return -1 since EOF is
1112      * not legal in this context. May return {@link #EMPTY_ESCAPE_SEQUENCE} to
1113      * indicate a zero-length sequence such as BS-NL.
1114      * @throws UnexpectedEofException if the EOF is encountered in the middle
1115      * of the escape sequence.
1116      * */
1117     public static int readEscapedCharacter(PushbackReader r, boolean inClob)
1118         throws IOException, UnexpectedEofException
1119     {
1120         //    \\ The backslash character
1121         //    \0 The character 0 (nul terminator char, for some)
1122         //    \xhh The character with hexadecimal value 0xhh
1123         //    \ u hhhh The character with hexadecimal value 0xhhhh
1124         //    \t The tab character ('\u0009')
1125         //    \n The newline (line feed) character ('\ u 000A')
1126         //    \v The vertical tab character ('\u000B')
1127         //    \r The carriage-return character ('\ u 000D')
1128         //    \f The form-feed character ('\ u 000C')
1129         //    \a The alert (bell) character ('\ u 0007')
1130         //    \" The double quote character
1131         //    \' The single quote character
1132         //    \? The question mark character
1133 
1134         int c2 = 0, c = r.read();
1135         switch (c) {
1136             case -1:
1137                 throw new UnexpectedEofException();
1138             case 't':  return '\t';
1139             case 'n':  return '\n';
1140             case 'v':  return '\u000B';
1141             case 'r':  return '\r';
1142             case 'f':  return '\f';
1143             case 'b':  return '\u0008';
1144             case 'a':  return '\u0007';
1145             case '\\': return '\\';
1146             case '\"': return '\"';
1147             case '\'': return '\'';
1148             case '/':  return '/';
1149             case '?':  return '?';
1150 
1151             case 'U':
1152                 // Expecting 8 hex digits
1153                 if (inClob) {
1154                     throw new IonException("Unicode escapes \\U not allowed in clob");
1155                 }
1156                 c = readDigit(r, 16, true);
1157                 if (c < 0) break;  // TODO throw UnexpectedEofException
1158                 c2 = c << 28;
1159                 c = readDigit(r, 16, true);
1160                 if (c < 0) break;
1161                 c2 += c << 24;
1162                 c = readDigit(r, 16, true);
1163                 if (c < 0) break;
1164                 c2 += c << 20;
1165                 c = readDigit(r, 16, true);
1166                 if (c < 0) break;
1167                 c2 += c << 16;
1168                 // ... fall through...
1169             case 'u':
1170                 // Expecting 4 hex digits
1171                 if (inClob) {
1172                     throw new IonException("Unicode escapes \\u not allowed in clob");
1173                 }
1174                 c = readDigit(r, 16, true);
1175                 if (c < 0) break;
1176                 c2 += c << 12;
1177                 c = readDigit(r, 16, true);
1178                 if (c < 0) break;
1179                 c2 += c << 8;
1180                 // ... fall through...
1181             case 'x':
1182                 // Expecting 2 hex digits
1183                 c = readDigit(r, 16, true);
1184                 if (c < 0) break;
1185                 c2 += c << 4;
1186                 c = readDigit(r, 16, true);
1187                 if (c < 0) break;
1188                 return c2 + c;
1189 
1190             case '0':
1191                 return 0;
1192             case '\n':
1193                 return EMPTY_ESCAPE_SEQUENCE;
1194             default:
1195                 break;
1196         }
1197         throw new IonException("invalid escape sequence \"\\"
1198                                + (char) c + "\" [" + c + "]");
1199     }

1252     public Type readNumber(int c) throws IOException {
1253         // clear out our string buffer
1254         resetValue();
1255 
1256         // process the initial sign character, if its there.
1257         boolean explicitPlusSign = false;
1258         switch (c) {
1259         case '-':
1260             value.append((char)c);
1261             c = this.read();
1262             t = Type.constNegInt;
1263             this.numberType = NumberType.NT_NEGINT;
1264             this.isNegative = true;
1265             break;
1266         case '+':
1267             // we eat the plus sign, but remember we saw one
1268             explicitPlusSign = true;
1269             c = this.read();
1270             t = Type.constPosInt;
1271             this.isNegative = false;
1272             this.numberType = NumberType.NT_POSINT;
1273             break;
1274         default:
1275             t = Type.constPosInt;
1276             this.numberType = NumberType.NT_POSINT;
1277             this.isNegative = false;
1278             break;
1279         }
1280 
1281         // process the initial digit.  Caller has checked that it's a digit.
1282         assert isDigit(c, 10);
1283         value.append((char)c);
1284         boolean isZero = (c == '0');
1285         boolean leadingZero = isZero;
1286 
1287         // process the next character after the initial digit.
1288         switch((c = this.read())) {
1289         case 'x':
1290         case 'X':
1291             if (!isZero) {
1292                 throw new IonException("badly formed number encountered at " + position());
1293             }
1294             this.numberType = NumberType.NT_HEX;
1295             // We don't need to append the char because we're going to wipe
1296             // the value anyway to accumulate just the hex digits.
1297             return scanHexNumber();
1298         case '.':
1299         case 'd':
1300         case 'D':
1301             t = Type.constDecimal;
1302             this.numberType = NumberType.NT_DECIMAL;
1303             value.append((char)c);
1304             break;
1305         case 'e':
1306         case 'E':
1307             t = Type.constFloat;
1308             this.numberType = NumberType.NT_FLOAT;
1309             value.append((char)c);
1310             break;
1311         default:
1312             if (!isDigit(c, 10)) {
1313                 checkAndUnreadNumericStopper(c);
1314                 if (isZero && NumberType.NT_NEGINT.equals(this.numberType)) {
1315                     t = Type.constPosInt;
1316                     this.numberType = NumberType.NT_POSINT;
1317                 }
1318                 return t;
1319             }
1320             value.append((char)c);
1321             isZero &= (c == '0');
1322             break;
1323         }
1324 
1325         // see if see can continue find leading digits
1326         if (NumberType.NT_NEGINT.equals(numberType)
1327          || NumberType.NT_POSINT.equals(numberType)
1328         ) {
1329             // We've now scanned at least two digits.
1330             // read in the remaining whole number digits
1331             for (;;) {
1332                 c = this.read();
1333                 if (!isDigit(c, 10)) break;
1334                 value.append((char)c);
1335                 isZero &= (c == '0');
1336             }
1337 
1338             // and see what we ran aground on ..
1339             switch (c) {
1340             case '.':
1341             case 'd':
1342             case 'D':
1343                 if (leadingZero) {
1344                     throw new IonException(position() + ": Invalid leading zero on numeric");
1345                 }
1346                 t = Type.constDecimal;
1347                 this.numberType = NumberType.NT_DECIMAL;
1348                 value.append((char)c);
1349                 break;
1350             case 'e':
1351             case 'E':
1352                 if (leadingZero) {
1353                     throw new IonException(position() + ": Invalid leading zero on numeric");
1354                 }
1355                 t = Type.constFloat;
1356                 this.numberType = NumberType.NT_FLOAT;
1357                 value.append((char)c);
1358                 break;
1359             case 'T':  // same as '-' it's a timestamp
1360             case '-':
1361                 if (NumberType.NT_POSINT.equals(this.numberType) && !explicitPlusSign) {
1362                     return scanTimestamp(c);
1363                 }
1364                 // otherwise fall through to "all other" processing
1365             default:
1366                 // Hit the end of our integer.  Make sure its a valid stopper.
1367                 checkAndUnreadNumericStopper(c);
1368 
1369                 if (leadingZero && !isZero) {
1370                     throw new IonException(position() + ": Invalid leading zero on numeric");
1371                 }
1372                 if (isZero && NumberType.NT_NEGINT.equals(this.numberType)) {
1373                     t = Type.constPosInt;
1374                     this.numberType = NumberType.NT_POSINT;
1375                 }
1376                 return t;
1377             }
1378         }
1379 
1380         // now we must have just decimal or float
1381         // if it's decimal we ended on a DOT, so get the trailing
1382         // (fractional) digits
1383         if (NumberType.NT_DECIMAL.equals(this.numberType)) {
1384             for (;;) {
1385                 c = this.read();
1386                 if (!isDigit(c, 10)) break;
1387                 value.append((char)c);
1388             }
1389             // and see what we ran aground on this time..
1390             switch (c) {
1391             case '-':
1392             case '+':
1393                 // We'll handle exponent below.
1394                 this.unread(c);
1395                 break;
1396             case 'e':
1397             case 'E':
1398                 // this is the only viable option now
1399                 t = Type.constFloat;
1400                 this.numberType = NumberType.NT_FLOAT;
1401                 value.append((char)c);
1402                 break;
1403             case 'd':
1404             case 'D':
1405                 // this is the only viable option now
1406                 assert t == Type.constDecimal;
1407                 assert NumberType.NT_DECIMAL.equals(this.numberType);
1408                 value.append((char)c);
1409                 break;
1410             default:
1411                 checkAndUnreadNumericStopper(c);
1412                 return t;
1413             }
1414         }
1415 
1416         // now we have the exponent part to read
1417         // reading the first character of the exponent, it can be
1418         // a sign character or a digit (this is the only place
1419         // the sign is valid) and we're for sure a float at this point
1420         switch ((c = this.read())) {
1421         case '-':
1422             value.append((char)c);
1423             break;
1424         case '+':
1425             // we eat the plus sign
1426             break;
1427         default:
1428             if (!isDigit(c, 10)) {
1429                 // this they said 'e' then they better put something valid after it!
1430                 throw new IonException("badly formed number encountered at " + position());
1431             }
1432             this.unread(c);
1433             // we'll re-read this in just a bit, but we had to check
1434             break;
1435         }
1436 
1437         // read in the remaining whole number digits and then quit
1438         for (;;) {
1439             c = this.read();
1440             if (!isDigit(c, 10)) break;
1441             value.append((char)c);
1442         }
1443 
1444         checkAndUnreadNumericStopper(c);
1445 
1446         return t;
1447     }

1449     Type scanHexNumber() throws IOException
1450     {
1451         boolean anydigits = false;
1452         int c;
1453 
1454         boolean isZero = true;
1455 
1456         // Leave the 0x in the buffer since we'll need it to parse the content.
1457         // first get clear out our string buffer, we're starting over
1458         value.setLength(0);  // resetValue() wipes out too much
1459 
1460         // read in the remaining whole number digits and then quit
1461         for (;;) {
1462             c = this.read();
1463             if (!isDigit(c, 16)) break;
1464             anydigits = true;
1465             isZero &= (c == '0');
1466             value.append((char)c);
1467         }
1468         if (!anydigits) {
1469             throw new IonException("badly formed hexadecimal number encountered at " + position());
1470         }
1471 
1472         checkAndUnreadNumericStopper(c);
1473         if (isZero && t == Type.constNegInt) {
1474             t = Type.constPosInt;
1475         }
1476 
1477         // as long as we saw any digit, we're good
1478         return t;
1479     }

1482      * Scans a timestamp after reading <code>yyyy-</code>.
1483      *
1484      * We can be a little lenient here since the result will be reparsed and
1485      * validated more thoroughly by {@link Timestamp#valueOf(CharSequence)}.
1486      *
1487      * @param c the last character scanned; must be <code>'-'</code>.
1488      * @return {@link Type#constTime}
1489      */
1490     Type scanTimestamp(int c) throws IOException {
1491 
1492 endofdate:
1493         for (;;) {  // fake for loop to create a label we can jump out of,
1494                     // because 4 or 5 levels of nested if's is just ugly
1495 
1496             // at this point we will have read leading digits and exactly 1 dash
1497             // in other words, we'll have read the year
1498             if (c == 'T') {
1499                 // yearT is a valid timestamp value
1500                 value.append((char)c);
1501                 c = this.read(); // because we'll unread it before we return
1502                 break endofdate;
1503             }
1504             if (c != '-') {
1505                 // not a dash or a T after the year - so this is a bad value
1506                 throw new IllegalStateException("invalid timestamp, expecting a dash here at " + this.position());
1507             }
1508 
1509             // append the dash and then read the month field
1510             value.append((char)c); // so append it, because we haven't already
1511             c = readDigits(2, "month");
1512             if (c == 'T') {
1513                 // year-monthT is a valid timestamp value
1514                 value.append((char)c);
1515                 c = this.read(); // because we'll unread it before we return
1516                 break endofdate;
1517             }
1518             if (c != '-') {
1519                 // if the month isn't followed by a dash or a T it's an invalid month
1520                 throw new IonException("invalid timestamp, expecting month at " + this.position());
1521             }
1522 
1523             // append the dash and read the day (or day-of-month) field
1524             value.append((char)c);
1525             c = readDigits(2, "day of month");
1526             if (c == 'T') {
1527 
1528 check4timezone:
1529                 for (;;) { // another fake label/ for=goto
1530 
1531                     // attach the 'T' to the value we're collecting
1532                     value.append((char)c);
1533 
1534                     // we're going to "watch" how many digits we read in the hours
1535                     // field.  It's 0 that's actually ok, since we can end at the
1536                     // 'T' we just read
1537                     int length_before_reading_hours = value.length();
1538                     // so read the hours
1539                     c = readDigits(2, "hours");
1540                     if (length_before_reading_hours == value.length()) {
1541                         // FIXME I don't think there should be a timezone here
1542                         break check4timezone;
1543                     }
1544                     if (c != ':') {
1545                         throw new IonException("invalid timestamp, expecting hours at " + this.position());
1546                     }
1547                     value.append((char)c);
1548                     // so read the minutes
1549                     c = readDigits(2, "minutes");
1550                     if (c != ':') {
1551                         if (c == '-' || c == '+' || c == 'Z') {
1552                             break check4timezone;
1553                         }
1554                         break endofdate;
1555                     }
1556                     value.append((char)c);
1557                     // so read the seconds
1558                     c = readDigits(2, "seconds");
1559                     if (c != '.') {
1560                         if (c == '-' || c == '+' || c == 'Z') {
1561                             break check4timezone;
1562                         }
1563                         break endofdate;
1564                     }
1565                     value.append((char)c);
1566                     // so read the fractional seconds
1567                     c = readDigits(32, "fractional seconds");
1568                     break check4timezone;
1569                 }//check4timezone
1570 
1571                 // now check to see if it's a timezone offset we're looking at
1572                 if (c == '-' || c == '+') {
1573                     value.append((char)c);
1574                     // so read the timezone offset
1575                     c = readDigits(2, "timezone offset");
1576                     if (c != ':') break endofdate;
1577                     value.append((char)c);
1578                     c = readDigits(2, "timezone offset");
1579                 }
1580                 else if (c == 'Z') {
1581                     value.append((char)c);
1582                     c = this.read(); // because we'll unread it before we return
1583                 }
1584             }
1585             break endofdate;
1586         }//endofdate
1587 
1588         checkAndUnreadNumericStopper(c);
1589 
1590         return Type.constTime;
1591     }

1593     int readDigits(int limit, String field) throws IOException {
1594         int c, len = 0;
1595         // read in the remaining whole number digits and then quit
1596         for (;;) {
1597             c = this.read();
1598             if (!isDigit(c, 10)) break;
1599             len++;
1600             if (len > limit) {
1601                 throw new IonException("invalid format " + field + " too long at " + this.position());
1602             }
1603             value.append((char)c);
1604         }
1605         return c;
1606     }
