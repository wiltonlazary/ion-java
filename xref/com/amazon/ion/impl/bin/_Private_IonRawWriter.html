16  package com.amazon.ion.impl.bin;
17  
18  import com.amazon.ion.IonWriter;
19  import java.io.IOException;
20  
21  /**
22   * An {@link IonWriter} with no symbol table management.
23   *
24   * @deprecated This is a private API subject to change without notice.
25   */
26  @Deprecated
27  public interface _Private_IonRawWriter extends IonWriter
28  {
29      /**
30       * Sets the current field name to the given symbol ID. It is up to the
31       * caller to make sure the given symbol ID has a mapping in the current
32       * context's symbol table. The pending field name symbol is cleared when the
33       * current value is written via stepIn() or one of the write*() methods.
34       * @param sid - a symbol ID
35       */
36      public void setFieldNameSymbol(int sid);
37  
38      /**
39       * Sets the full list of pending annotations to the given symbol IDs. Any
40       * pending annotation symbol IDs are cleared. It is up to the caller to make
41       * sure the given symbol IDs have mappings in the current context's symbol
42       * table. The contents of the given array are copied into this writer, so
43       * the caller does not need to preserve the array. The list of pending
44       * annotation symbol IDs is cleared when the current value is written via
45       * stepIn() or one of the write*() methods.
46       * @param sids - symbol IDs representing the annotation symbols for the
47       *               current value
48       */
49      public void setTypeAnnotationSymbols(int... sids);
50  
51      /**
52       * Adds the given symbol ID to the list of pending annotation symbol IDs. It
53       * is up to the caller to make sure the given symbol ID has a mapping in the
54       * current context's symbol table. The list of pending annotation symbol IDs
55       * is cleared when the current value is written via stepIn() or one of the
56       * write*() methods.
57       * @param sid - a symbol ID
58       */
59      public void addTypeAnnotationSymbol(int sid);
60  
61      /**
62       * Directly write the given symbol ID to represent a symbol value. It is up
63       * to the caller to make sure the given symbol ID has a mapping in the
64       * current context's symbol table.
65       * @param sid - a symbol ID
66       * @throws IOException
67       */
68      public void writeSymbolToken(int sid) throws IOException;
69  
70      /**
71       * Writes a portion of the byte array out as an IonString value.  This
72       * copies the portion of the byte array that is written.
73       *
74       * @param data well-formed UTF-8-encoded bytes to be written.
75       * May be {@code null} to represent {@code null.string}.
76       * @param offset offset of the first byte in value to write
77       * @param length number of bytes to write from value
78       * @see IonWriter#writeClob(byte[], int, int)
79       * @see IonWriter#writeBlob(byte[], int, int)
80       */
81      public void writeString(byte[] data, int offset, int length)
82          throws IOException;
83  
84  }
