26  public class ReaderCompare
27      extends Assert
28  {
29      public static void compare(IonReader it1, IonReader it2) {
30          while (hasNext(it1, it2)) {
31              IonType t1 = it1.getType();
32              IonType t2 = it2.getType();
33  
34              if ((t1 != t2) && (t1 == null || t2 == null || !t1.equals(t2))) {
35                  assertEquals("ion type", t1, t2);
36              }
37              if (t1 == null) break;
38  
39              if (it1.isInStruct()) {
40                  compareFieldNames(it1, it2);
41              }
42              compareAnnotations(it1, it2);
43  
44              boolean isNull = it1.isNullValue();
45              assertEquals(isNull, it2.isNullValue());
46  
47              switch (t1) {
48                  case NULL:
49                      assertTrue(it1.isNullValue());
50                      assertTrue(it2.isNullValue());
51                      break;
52                  case BOOL:
53                  case INT:
54                  case FLOAT:
55                  case DECIMAL:
56                  case TIMESTAMP:
57                  case STRING:
58                  case SYMBOL:
59                  case BLOB:
60                  case CLOB:
61                      compareScalars(t1, isNull, it1, it2);
62                      break;
63                  case STRUCT:
64                  case LIST:
65                  case SEXP:
66                      it1.stepIn();
67                      it2.stepIn();
68                      compare(it1, it2);
69                      it1.stepOut();
70                      it2.stepOut();
71                      break;
72  
73                  default:
74                      throw new IllegalStateException();
75              }
76          }
77          assertFalse(hasNext(it1, it2));
78      }

81      public static boolean hasNext(IonReader it1, IonReader it2)
82      {
83          boolean more;
84          more = (it1.next() != null);
85          assertEquals("next results don't match", more, it2.next() != null);
86  
87          if (!more) {
88              assertEquals(null, it1.next());
89              assertEquals(null, it2.next());
90          }
91  
92          return more;
93      }
94  
95      public static void compareNonNullStrings(String what, String s1, String s2)
96      {
97          assertNotNull(what, s1);
98          assertNotNull(what, s2);
99          assertEquals(what, s1, s2);
100     }

102     public static void compareFieldNames(IonReader r1, IonReader r2) {
103         SymbolToken tok1 = r1.getFieldNameSymbol();
104         SymbolToken tok2 = r1.getFieldNameSymbol();
105         String fn = tok1.getText();
106         assertEquals(fn, tok2.getText());
107 
108         if (fn != null) {
109             String f1 = r1.getFieldName();
110             String f2 = r2.getFieldName();
111             compareNonNullStrings("field name", fn, f1);
112             compareNonNullStrings("field name", fn, f2);
113         }
114     }
115 
116     public static void compareAnnotations(IonReader it1, IonReader it2) {
117         SymbolToken[] syms1 = it1.getTypeAnnotationSymbols();
118         SymbolToken[] syms2 = it2.getTypeAnnotationSymbols();
119 
120         IonAssert.assertSymbolEquals("annotation", syms1, syms2);
121     }

123     public static void compareScalars(IonType t, boolean isNull,
124                                       IonReader it1, IonReader it2) {
125         switch (t) {
126             case BOOL:
127                 if (!isNull) assertEquals(it1.booleanValue(), it2.booleanValue());
128                 break;
129             case INT:
130                 assertEquals(it1.bigIntegerValue(), it2.bigIntegerValue());
131                 break;
132             case FLOAT:
133             {
134                 if (! isNull)
135                 {
136                     double v1 = it1.doubleValue();
137                     double v2 = it2.doubleValue();
138                     assertEquals(v1, v2, 0);
139                     // The last param is a delta, and we want exact match.
140                 }
141                 break;
142             }
143             case DECIMAL:
144             {
145                 BigDecimal dec1 = it1.bigDecimalValue();
146                 BigDecimal dec2 = it2.bigDecimalValue();
147                 IonTestCase.assertPreciselyEquals(dec1, dec2);
148                 dec1 = it1.decimalValue();
149                 dec2 = it2.decimalValue();
150                 IonTestCase.assertPreciselyEquals(dec1, dec2);
151                 // TODO also test double, long, int etc.
152                 break;
153             }
154             case TIMESTAMP:
155             {
156                 Timestamp t1 = it1.timestampValue();
157                 Timestamp t2 = it2.timestampValue();
158                 assertEquals(t1, t2);
159                 break;
160             }
161             case STRING:
162             {
163                 String s1 = it1.stringValue();
164                 String s2 = it2.stringValue();
165                 assertEquals(s1, s2);
166                 break;
167             }
168             case SYMBOL:
169             {
170                 SymbolToken tok1 = it1.symbolValue();
171                 SymbolToken tok2 = it2.symbolValue();
172                 if (isNull)
173                 {
174                     assertNull(tok1);
175                     assertNull(tok2);
176                 }
177                 else if (tok1.getText() == null || tok2.getText() == null)
178                 {
179                     assertEquals("sids", tok1.getSid(), tok2.getSid());
180                 }
181                 else
182                 {
183                     String s1 = tok1.getText();
184                     String s2 = tok2.getText();
185                     assertEquals(s1, s2);
186                 }
187                 break;
188             }
189             case BLOB:
190             case CLOB:
191             {
192                 if (!isNull)
193                 {
194                     byte[] b1 = it1.newBytes();
195                     byte[] b2 = it2.newBytes();
196                     assert b1 != null && b2 != null;
197                     assert b1.length == b2.length;
198                     for (int ii=0; ii<b1.length; ii++) {
199                         byte v1 = b1[ii];
200                         byte v2 = b2[ii];
201                         assertEquals(v1, v2);
202                     }
203                 }
204                 break;
205             }
206             default:
207                 throw new IllegalStateException("iterated to a type that's not expected");
208         }
209     }
