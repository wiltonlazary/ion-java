28      /** A double that's too big for a float */
29      public static final double A_DOUBLE = 1D + Float.MAX_VALUE;
30  
31      public static void checkNullFloat(IonFloat value)
32      {
33          assertSame(IonType.FLOAT, value.getType());
34          assertTrue("isNullValue is false", value.isNullValue());
35          assertFalse(value.isNumericValue());
36  
37          try
38          {
39              value.floatValue();
40              fail("Expected NullValueException");
41          }
42          catch (NullValueException e) { }
43  
44          try
45          {
46              value.doubleValue();
47              fail("Expected NullValueException");
48          }
49          catch (NullValueException e) { }
50  
51          assertNull("bigDecimalValue() isn't null", value.bigDecimalValue());
52      }
53  
54  
55      public void modifyFloat(IonFloat value)
56      {
57          float  fVal = 123.45F;
58  
59          value.setValue(fVal);
60          assertEquals(fVal, value.floatValue());
61          assertEquals((double) fVal, value.doubleValue());
62          assertEquals(fVal, value.bigDecimalValue().floatValue());
63          assertTrue(value.isNumericValue());
64  
65          value.setValue(A_DOUBLE);
66          assertEquals(A_DOUBLE, value.doubleValue());
67          assertEquals(A_DOUBLE, value.bigDecimalValue().doubleValue());
68          assertTrue(value.isNumericValue());
69  
70          value.setValue(null);
71          checkNullFloat(value);
72      }

126     @Test
127     public void testParsingSpecialFloats()
128     {
129         IonFloat value = (IonFloat) oneValue("nan");
130         checkNan(value);
131 
132         value = (IonFloat) oneValue("+inf");
133         checkPosInf(value);
134 
135         value = (IonFloat) oneValue("-inf");
136         checkNegInf(value);
137     }

139     @Test
140     public void testCreatingSpecialFloats()
141     {
142         IonFloat value = system().newFloat(Double.NaN);
143         checkNan(value);
144 
145         value = system().newFloat(Double.POSITIVE_INFINITY);
146         checkPosInf(value);
147 
148         value = system().newFloat(Double.NEGATIVE_INFINITY);
149         checkNegInf(value);
150 
151         value.setValue(Float.NaN);
152         checkNan(value);
153 
154         value.setValue(Float.POSITIVE_INFINITY);
155         checkPosInf(value);
156 
157         value.setValue(Float.NEGATIVE_INFINITY);
158         checkNegInf(value);
159 
160         value.setValue(Double.NaN);
161         checkNan(value);
162 
163         value.setValue(Double.POSITIVE_INFINITY);
164         checkPosInf(value);
165 
166         value.setValue(Double.NEGATIVE_INFINITY);
167         checkNegInf(value);
168     }

170     public void checkNan(IonFloat actual)
171     {
172         checkSpecial(Float.NaN, Double.NaN, "nan", actual);
173     }
174 
175     public void checkPosInf(IonFloat actual)
176     {
177         checkSpecial(Float.POSITIVE_INFINITY, Double.POSITIVE_INFINITY, "+inf",
178                      actual);
179         checkSpecial(Float.POSITIVE_INFINITY, Double.POSITIVE_INFINITY, "+inf",
180                      actual.clone());
181     }
182 
183     public void checkNegInf(IonFloat actual)
184     {
185         checkSpecial(Float.NEGATIVE_INFINITY, Double.NEGATIVE_INFINITY, "-inf",
186                      actual);
187         checkSpecial(Float.NEGATIVE_INFINITY, Double.NEGATIVE_INFINITY, "-inf",
188                      actual.clone());
189     }

191     public void checkSpecial(float expectedFloat,
192                              double expectedDouble,
193                              String expectedText,
194                              IonFloat actual)
195     {
196         assertEquals(expectedFloat, actual.floatValue());
197         assertEquals(expectedDouble, actual.doubleValue());
198         assertEquals(expectedText, actual.toString());
199         assertFalse(actual.isNumericValue());
200 
201         try
202         {
203             actual.bigDecimalValue();
204             fail("expected exception");
205         }
206         catch (NumberFormatException e) { }
207     }
