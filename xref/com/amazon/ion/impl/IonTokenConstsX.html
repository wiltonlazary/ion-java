213     private final static int[] makeHexValueArray() {
214         int[] hex = new int[256];
215         for (int ii=0; ii<256; ii++) {
216             hex[ii] = -1;
217         }
218         for (int ii='0'; ii<='9'; ii++) {
219             hex[ii] = ii - '0';
220         }
221         for (int ii='a'; ii<='f'; ii++) {
222             hex[ii] = ii - 'a' + 10;
223         }
224         for (int ii='A'; ii<='F'; ii++) {
225             hex[ii] = ii - 'A' + 10;
226         }
227         return hex;
228     }

294     private static final int escapeCharactersValues[] = makeEscapeCharacterValuesArray();
295     private static final int[] makeEscapeCharacterValuesArray() {
296         int[] values = new int[256];
297         for (int ii=0; ii<256; ii++) {
298             values[ii] = ESCAPE_NOT_DEFINED;
299         }
300         values['0'] = 0;        //    \u0000  \0  alert NUL
301         values['a'] = 7;        //    \u0007  \a  alert BEL
302         values['b'] = 8;        //    \u0008  \b  backspace BS
303         values['t'] = 9;        //    \u0009  \t  horizontal tab HT
304         values['n'] = '\n';     //    \ u000A  \ n  linefeed LF
305         values['f'] = 0x0c;     //    \u000C  \f  form feed FF
306         values['r'] = '\r';     //    \ u000D  \ r  carriage return CR
307         values['v'] = 0x0b;     //    \u000B  \v  vertical tab VT
308         values['"'] = '"';      //    \u0022  \"  double quote
309         values['\''] = '\'';    //    \u0027  \'  single quote
310         values['?'] = '?';      //    \u003F  \?  question mark
311         values['\\'] = '\\';    //    \u005C  \\  backslash
312         values['/'] = '/';      //    \u002F  \/  forward slash nothing  \NL  escaped NL expands to nothing
313         values['\n'] = ESCAPE_REMOVES_NEWLINE;  // slash-new line the new line eater
314         values['\r'] = ESCAPE_REMOVES_NEWLINE2;  // slash-new line the new line eater
315         values['x'] = ESCAPE_HEX; //    any  \xHH  2-digit hexadecimal unicode character equivalent to \ u00HH
316         values['u'] = ESCAPE_LITTLE_U; //    any  \ uHHHH  4-digit hexadecimal unicode character
317         values['U'] = ESCAPE_BIG_U;
318         return values;

387     public static String escapeSequence(int c) {
388         if (c >= 0 || c <= 0x10FFFF) {
389             if (c < 128) {
390                 return escapeCharacterImage[c];
391             }
392             if (c < 0xFFFF) {
393                 String short_hex = Integer.toHexString(c);
394                 int    short_len = short_hex.length();
395                 if (short_len < 4) {
396                     short_hex = "0000".substring(short_len);
397                 }
398                 return "\\u"+short_hex;
399             }
400             if (c < 0xFFFF) {
401                 String long_hex = Integer.toHexString(c);
402                 int    long_len = long_hex.length();
403                 if (long_len < 4) {
404                     long_hex = "00000000".substring(long_len);
405                 }
406                 return "\\U"+long_hex;
407             }
408         }
409         throw new IllegalArgumentException("the value "+c+" isn't a valid character");
410     }

519     static public int keyword(CharSequence word, int start_word, int end_word)
520     {
521         int c = word.charAt(start_word);
522         int len = end_word - start_word; // +1 but we build that into the constants below
523         switch (c) {
524         case '$':
525             if (len > 1) {
526                 for (int i = start_word + 1; i < end_word; i++) {
527                     if (! isDigit(word.charAt(i))) return -1;
528                 }
529                 return KEYWORD_sid;
530             }
531             return -1;
532         case 'b':
533             if (len == 4) {
534                 if (word.charAt(start_word+1) == 'o'
535                  && word.charAt(start_word+2) == 'o'
536                  && word.charAt(start_word+3) == 'l'
537                 ) {
538                     return KEYWORD_BOOL;
539                 }
540                 if (word.charAt(start_word+1) == 'l'
541                  && word.charAt(start_word+2) == 'o'
542                  && word.charAt(start_word+3) == 'b'
543                 ) {
544                     return KEYWORD_BLOB;
545                 }
546             }
547             return -1;
548         case 'c':
549             if (len == 4) {
550                 if (word.charAt(start_word+1) == 'l'
551                  && word.charAt(start_word+2) == 'o'
552                  && word.charAt(start_word+3) == 'b'
553                 ) {
554                     return KEYWORD_CLOB;
555                 }
556             }
557             return -1;
558         case 'd':
559             if (len == 7) {
560                 if (word.charAt(start_word+1) == 'e'
561                  && word.charAt(start_word+2) == 'c'
562                  && word.charAt(start_word+3) == 'i'
563                  && word.charAt(start_word+4) == 'm'
564                  && word.charAt(start_word+5) == 'a'
565                  && word.charAt(start_word+6) == 'l'
566                 ) {
567                     return KEYWORD_DECIMAL;
568                 }
569             }
570             return -1;
571         case 'f':
572             if (len == 5) {
573                 if (word.charAt(start_word+1) == 'a'
574                  && word.charAt(start_word+2) == 'l'
575                  && word.charAt(start_word+3) == 's'
576                  && word.charAt(start_word+4) == 'e'
577                 ) {
578                     return KEYWORD_FALSE;
579                 }
580                 if (word.charAt(start_word+1) == 'l'
581                  && word.charAt(start_word+2) == 'o'
582                  && word.charAt(start_word+3) == 'a'
583                  && word.charAt(start_word+4) == 't'
584                 ) {
585                     return KEYWORD_FLOAT;
586                 }
587             }
588             return -1;
589         case 'i':
590             if (len == 3) {
591                 if (word.charAt(start_word+1) == 'n') {
592                     if (word.charAt(start_word+2) == 't') {
593                         return KEYWORD_INT;
594                     }
595                 }
596             }
597             return -1;
598         case 'l':
599             if (len == 4) {
600                 if (word.charAt(start_word+1) == 'i'
601                  && word.charAt(start_word+2) == 's'
602                  && word.charAt(start_word+3) == 't'
603                 ) {
604                     return KEYWORD_LIST;
605                 }
606             }
607             return -1;
608         case 'n':
609             if (len == 4) {
610                 if (word.charAt(start_word+1) == 'u'
611                  && word.charAt(start_word+2) == 'l'
612                  && word.charAt(start_word+3) == 'l'
613                 ) {
614                     return KEYWORD_NULL;
615                 }
616             }
617             else if (len == 3) {
618                 if (word.charAt(start_word+1) == 'a'
619                     && word.charAt(start_word+2) == 'n'
620                    ) {
621                        return KEYWORD_NAN;
622                    }
623             }
624             return -1;
625         case 's':
626             if (len == 4) {
627                 if (word.charAt(start_word+1) == 'e'
628                  && word.charAt(start_word+2) == 'x'
629                  && word.charAt(start_word+3) == 'p'
630                 ) {
631                     return KEYWORD_SEXP;
632                 }
633             }
634             else if (len == 6) {
635                 if (word.charAt(start_word+1) == 't'
636                  && word.charAt(start_word+2) == 'r'
637                 ) {
638                     if (word.charAt(start_word+3) == 'i'
639                      && word.charAt(start_word+4) == 'n'
640                      && word.charAt(start_word+5) == 'g'
641                     ) {
642                         return KEYWORD_STRING;
643                     }
644                     if (word.charAt(start_word+3) == 'u'
645                      && word.charAt(start_word+4) == 'c'
646                      && word.charAt(start_word+5) == 't'
647                     ) {
648                         return KEYWORD_STRUCT;
649                     }
650                     return -1;
651                 }
652                 if (word.charAt(start_word+1) == 'y'
653                  && word.charAt(start_word+2) == 'm'
654                  && word.charAt(start_word+3) == 'b'
655                  && word.charAt(start_word+4) == 'o'
656                  && word.charAt(start_word+5) == 'l'
657                 ) {
658                     return KEYWORD_SYMBOL;
659                 }
660             }
661             return -1;
662         case 't':
663             if (len == 4) {
664                 if (word.charAt(start_word+1) == 'r'
665                  && word.charAt(start_word+2) == 'u'
666                  && word.charAt(start_word+3) == 'e'
667                 ) {
668                     return KEYWORD_TRUE;
669                 }
670             }
671             else if (len == 9) {
672                 if (word.charAt(start_word+1) == 'i'
673                  && word.charAt(start_word+2) == 'm'
674                  && word.charAt(start_word+3) == 'e'
675                  && word.charAt(start_word+4) == 's'
676                  && word.charAt(start_word+5) == 't'
677                  && word.charAt(start_word+6) == 'a'
678                  && word.charAt(start_word+7) == 'm'
679                  && word.charAt(start_word+8) == 'p'
680                 ) {
681                     return KEYWORD_TIMESTAMP;
682                 }
683             }
684             return -1;
685         default:
686             return -1;
687         }
688     }

691     KW_BIT_BLOB       = 0x0001;
692     KW_BIT_BOOL       = 0x0002;
693     KW_BIT_CLOB       = 0x0004;
694     KW_BIT_DECIMAL    = 0x0008;
695     KW_BIT_FLOAT      = 0x0010;
696     KW_BIT_INT        = 0x0020;
697     KW_BIT_LIST       = 0x0030;
698     KW_BIT_NULL       = 0x0080;
699     KW_BIT_SEXP       = 0x0100;
700     KW_BIT_STRING     = 0x0200;
701     KW_BIT_STRUCT     = 0x0400;
702     KW_BIT_SYMBOL     = 0x0800;
703     KW_BIT_TIMESTAMP  = 0x1000;
704     datagram is not included (since these are types that can appear as a null type)
705 
706     A  1    F  6    N 11    T 16
707     B  2    G  7    O 12    U 17
708     C  3    I  8    P 13    X 18
709     D  4    L  9    R 14    Y 19
710     E  5    M 10    S 15
711 
712     */
713     // this array is [pos][letter_idx] - if this letter is
714     // valid in this position then all the type names that
715     // contain the letter at that position are on.  Each
716     // keyword is assigned a bit value.
717     // however due to the double index expense the index
718     // is represented by a 1 dim int array and fn's used
719     // to access it (since inline-ing final static small methods
720     // is easy for the compiler)
721     static final int  TN_MAX_NAME_LENGTH = "TIMESTAMP".length() + 1; // so anything too long will be 0
722     static final int  TN_LETTER_MAX_IDX = 19;
723 
724     static final int  KW_BIT_BLOB       = 0x0001;
725     static final int  KW_BIT_BOOL       = 0x0002;
726     static final int  KW_BIT_CLOB       = 0x0004;
727     static final int  KW_BIT_DECIMAL    = 0x0008;
728     static final int  KW_BIT_FLOAT      = 0x0010;
729     static final int  KW_BIT_INT        = 0x0020;
730     static final int  KW_BIT_LIST       = 0x0030;
731     static final int  KW_BIT_NULL       = 0x0080;
732     static final int  KW_BIT_SEXP       = 0x0100;
733     static final int  KW_BIT_STRING     = 0x0200;
734     static final int  KW_BIT_STRUCT     = 0x0400;
735     static final int  KW_BIT_SYMBOL     = 0x0800;
736     static final int  KW_BIT_TIMESTAMP  = 0x1000;
737     static final int  KW_ALL_BITS       = 0x1fff;
738 
739     static final int[] typeNameBits = new int[] {
740         KW_BIT_BLOB,      KW_BIT_BOOL,    KW_BIT_CLOB,
741         KW_BIT_DECIMAL,   KW_BIT_FLOAT,   KW_BIT_INT,
742         KW_BIT_LIST,      KW_BIT_NULL,    KW_BIT_SEXP,
743         KW_BIT_STRING,    KW_BIT_STRUCT,  KW_BIT_SYMBOL,
744         KW_BIT_TIMESTAMP
745     };

747     static final String[] typeNameNames = new String[] {
748         "blob",           "bool",         "clob",
749         "decimal",        "float",        "int",
750         "list",           "null",         "sexp",
751         "string",         "struct",       "symbol",
752         "timestamp"
753     };
754     static final int[] typeNameKeyWordIds = new int[] {
755         KEYWORD_BLOB,     KEYWORD_BOOL,   KEYWORD_CLOB,
756         KEYWORD_DECIMAL,  KEYWORD_FLOAT,  KEYWORD_INT,
757         KEYWORD_LIST,     KEYWORD_NULL,   KEYWORD_SEXP,
758         KEYWORD_STRING,   KEYWORD_STRUCT, KEYWORD_SYMBOL,
759         KEYWORD_TIMESTAMP
760     };

762     static final int[] TypeNameBitIndex = makeTypeNameBitIndex();
763     static final int[] makeTypeNameBitIndex() {
764         int[] bits = new int[TN_MAX_NAME_LENGTH*TN_LETTER_MAX_IDX];
765         int   typename_bit;
766         for (int tt=0; tt<typeNameNames.length; tt++) {
767             String typename = typeNameNames[tt];
768             typename_bit = typeNameBits[tt];
769             for (int ii=0; ii<typename.length(); ii++) {
770                 int c = typename.charAt(ii);
771                 int l = typeNameLetterIdx(c);
772                 assert(l > 0);
773                 typename_set_bit(bits, ii, l, typename_bit);
774             }
775         }
776         return bits;
777     }
778     private final static void typename_set_bit(int[] bits, int ii, int l, int typename_bit)
779     {
780         // bits[ii][l]
781         int idx = (ii * TN_LETTER_MAX_IDX) + l - 1;
782         bits[idx] |= typename_bit;
783     }
784     public final static int typeNameLetterIdx(int c) {
785         switch(c) {
786         case 'a': return  1;
787         case 'b': return  2;
788         case 'c': return  3;
789         case 'd': return  4;
790         case 'e': return  5;
791         case 'f': return  6;
792         case 'g': return  7;
793         case 'i': return  8;
794         case 'l': return  9;
795         case 'm': return 10;
796         case 'n': return 11;
797         case 'o': return 12;
798         case 'p': return 13;
799         case 'r': return 14;
800         case 's': return 15;
801         case 't': return 16;
802         case 'u': return 17;
803         case 'x': return 18;
804         case 'y': return 19;
805         default:  return -1;
806         }
807     }
808     public final static int typeNamePossibilityMask(int pos, int letter_idx) {
809         int idx = (pos * TN_LETTER_MAX_IDX) + letter_idx - 1;
810         int mask = TypeNameBitIndex[idx];
811         return mask;
812     }
813     // this can be faster but it's pretty unusual to be called.
814     public final static int typeNameKeyWordFromMask(int possible_names, int length) {
815         int kw = KEYWORD_unrecognized;
816         if (possible_names != IonTokenConstsX.KW_ALL_BITS) {
817             for (int ii=0; ii<typeNameBits.length; ii++) {
818                 int tb = typeNameBits[ii];
819                 if (tb == possible_names) {
820                     if (typeNameNames[ii].length() == length) {
821                         kw = typeNameKeyWordIds[ii];
822                     }
823                     break;
824                 }
825             }
826         }
827         return kw;
828     }
829     public static final String getNullImage(IonType type)
830     {
831         String nullimage = null;
832 
833         switch (type) {
834         case NULL:      nullimage = "null";           break;
835         case BOOL:      nullimage = "null.bool";      break;
836         case INT:       nullimage = "null.int";       break;
837         case FLOAT:     nullimage = "null.float";     break;
838         case DECIMAL:   nullimage = "null.decimal";   break;
839         case TIMESTAMP: nullimage = "null.timestamp"; break;
840         case SYMBOL:    nullimage = "null.symbol";    break;
841         case STRING:    nullimage = "null.string";    break;
842         case BLOB:      nullimage = "null.blob";      break;
843         case CLOB:      nullimage = "null.clob";      break;
844         case SEXP:      nullimage = "null.sexp";      break;
845         case LIST:      nullimage = "null.list";      break;
846         case STRUCT:    nullimage = "null.struct";    break;

853     public static final IonType getNullType(CharSequence s)
854     {
855         IonType type = null;
856         int     c, ii = 0;
857         boolean stop = false;
858         while (!stop && ii<s.length()) {
859             c = s.charAt(ii++);
860             switch (c) {
861             case ' ': case '\t': case '\r': case '\n':
862                 break;
863             case 'n':
864                 stop = true;
865                 break;
866             default:
867                 invalid_null_image(s);
868             }
869         }
870         if (ii>=s.length() || s.charAt(ii++) != 'u') invalid_null_image(s);
871         if (ii>=s.length() || s.charAt(ii++) != 'l') invalid_null_image(s);
872         if (ii>=s.length() || s.charAt(ii++) != 'l') invalid_null_image(s);
873         boolean dot = false;
874         while (!dot && ii<s.length()) {
875             c = s.charAt(ii++);
876             switch (c) {
877             case ' ': case '\t': case '\r': case '\n':
878                 break;
879             case '.':
880                 dot = true;
881                 break;
882             default:
883                 invalid_null_image(s);
884             }
885         }
886         if (dot) {
887             int kw = IonTokenConstsX.keyword(s, ii, s.length());
888             switch (kw) {
889                 case IonTokenConstsX.KEYWORD_NULL:
890                     type = IonType.NULL;
891                     break;
892                 case IonTokenConstsX.KEYWORD_BOOL:
893                     type = IonType.BOOL;
894                     break;
895                 case IonTokenConstsX.KEYWORD_INT:
896                     type = IonType.INT;
897                     break;
898                 case IonTokenConstsX.KEYWORD_FLOAT:
899                     type = IonType.FLOAT;
900                     break;
901                 case IonTokenConstsX.KEYWORD_DECIMAL:
902                     type = IonType.DECIMAL;
903                     break;
904                 case IonTokenConstsX.KEYWORD_TIMESTAMP:
905                     type = IonType.TIMESTAMP;
906                     break;
907                 case IonTokenConstsX.KEYWORD_SYMBOL:
908                     type = IonType.SYMBOL;
909                     break;
910                 case IonTokenConstsX.KEYWORD_STRING:
911                     type = IonType.STRING;
912                     break;
913                 case IonTokenConstsX.KEYWORD_CLOB:
914                     type = IonType.CLOB;
915                     break;
916                 case IonTokenConstsX.KEYWORD_BLOB:
917                     type = IonType.BLOB;
918                     break;
919                 case IonTokenConstsX.KEYWORD_STRUCT:
920                     type = IonType.STRUCT;
921                     break;
922                 case IonTokenConstsX.KEYWORD_LIST:
923                     type = IonType.LIST;
924                     break;
925                 case IonTokenConstsX.KEYWORD_SEXP:
926                     type = IonType.SEXP;
927                     break;
928                 default:
929                     invalid_null_image(s);
930             }
931         }
932 
933         while (ii<s.length()) {
934             c = s.charAt(ii++);
935             switch (c) {
936             case ' ': case '\t': case '\r': case '\n':
937                 break;
938             default:
939                 invalid_null_image(s);
940             }
941         }
942 
943         return type;
944     }
945     private static void invalid_null_image(CharSequence s) {
946         throw new CantConvertException("invalid image "+s.toString());
947     }
