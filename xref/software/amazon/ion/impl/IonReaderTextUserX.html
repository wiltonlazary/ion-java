33  /**
34   *    The text user reader add support for symbols and recognizes,
35   *    and consumes (and processes), the system values $ion_1_0 and
36   *    local symbol tables (tagged with $ion_symbol_table).
37   *
38   *    Should this materialize and "symbolate" all the symbol
39   *    values as they come through?  - No.
40   *
41   *    Probably if we want the symbol id's to be the same for this
42   *    reader as it is for other variants.  Hmmm, that's expensive
43   *    when you don't need it (which is most of the time).
44   *
45   *    This will not auto-populate a symbol table.  In the event
46   *    a symbol is a '$<digits>' symbol id symbol this will return
47   *    that value.  If the string is present in the current symbol
48   *    table it will return the id, which would be true if the
49   *    symbol is a system symbol or if there is a local symbol
50   *    table in the input stream.  Otherwise it return the
51   *    undefined symbol value.
52   */

54      extends IonReaderTextSystemX
55      implements PrivateReaderWriter

90      /**
91       * this looks forward to see if there is an upcoming value
92       * and if there is it returns true.  It may have to clean up
93       * any value that's partially complete (for example a
94       * collection whose annotation has been read and loaded
95       * but the use has chosen not to step into the collection).
96       *
97       * The user reader variant of hasNext also looks for system
98       * values to process.  System values are the Ion version
99       * marker ($ion_1_0) and local symbol tables.  If either of
100      * these is encountered the symbol table processing will be
101      * handled and the value will be "skipped".
102      *
103      * @return true if more data remains, false on eof
104      */
105     @Override
106     boolean hasNext()
107     {
108         boolean has_next = has_next_user_value();
109         return has_next;
110     }
111     private final boolean has_next_user_value()
112     {
113         // clear out our previous value
114         clear_system_value_stack();
115 
116         // changed to 'while' since consumed
117         // values will not be counted
118         while (!_has_next_called)
119         {
120             // first move to the next value regardless of whether
121             // it's a system value or a user value
122             has_next_raw_value();
123 
124             // system values are only at the datagram level
125             // we don't care about them if they're buried
126             // down in some other value - note that _value_type
127             // will be null at eof and on as yet undetermined
128             // numeric types (which are never system values)
129             if (_value_type != null && !isNullValue() && IonType.DATAGRAM.equals(getContainerType())) {
130                 switch (_value_type) {
131                 case STRUCT:
132                     if (_annotation_count > 0 && ION_SYMBOL_TABLE.equals(_annotations[0].getText())) {
133                         _symbols = _lstFactory.newLocalSymtab(_catalog,
134                                                               this,
135                                                               true);
136                         push_symbol_table(_symbols);
137                         _has_next_called = false;
138                     }
139                     break;
140                 case SYMBOL:
141                     if (_annotation_count == 0)
142                     {
143                         // $ion_1_0 is read as an IVM only if it is not annotated
144                         String version = symbolValue().getText();
145                         if (isIonVersionMarker(version))
146                         {
147                             if (ION_1_0.equals(version))
148                             {
149                                 if (_value_keyword != IonTokenConstsX.KEYWORD_sid)
150                                 {
151                                     symbol_table_reset();
152                                     push_symbol_table(_system_symtab);
153                                 }
154                                 _has_next_called = false;
155                             }
156                             else
157                             {
158                                 throw new UnsupportedIonVersionException(version);
159                             }
160                         }
161                     }
162                     break;
163                 default:
164                     break;
165                 }
166             }
167         }
168         return (_eof != true);
169     }

171     private static boolean isIonVersionMarker(String text)
172     {
173         return text != null && ION_VERSION_MARKER_REGEX.matcher(text).matches();
174     }

176     private final void symbol_table_reset()
177     {
178         IonType t = next();
179         assert( IonType.SYMBOL.equals(t) );
180         _symbols = _system_symtab;
181         return;
182     }

184     private void validateSymbolToken(SymbolToken symbol) {
185         if (symbol != null) {
186             if (symbol.getText() == null && symbol.getSid() > getSymbolTable().getMaxId()) {
187                 throw new UnknownSymbolException(symbol.getSid());
188             }
189         }

223     //  This code handles the skipped symbol table

230     private int _symbol_table_top = 0;
231     private SymbolTable">SymbolTable[] _symbol_table_stack = new SymbolTable[3]; // 3 is rare, IVM followed by a local sym tab with open content
232     private void clear_system_value_stack()
233     {
234         while (_symbol_table_top > 0) {
235             _symbol_table_top--;
236             _symbol_table_stack[_symbol_table_top] = null;
237         }
238     }
239     private void push_symbol_table(SymbolTable symbols)
240     {
241         assert(symbols != null);
242         if (_symbol_table_top >= _symbol_table_stack.length) {
243             int new_len = _symbol_table_stack.length * 2;
244             SymbolTabletml#SymbolTable">SymbolTable[] temp = new SymbolTable[new_len];
245             System.arraycopy(_symbol_table_stack, 0, temp, 0, _symbol_table_stack.length);
246             _symbol_table_stack = temp;
247         }
248         _symbol_table_stack[_symbol_table_top++] = symbols;
249     }
250     @Override
251     public SymbolTable pop_passed_symbol_table()
252     {
253         if (_symbol_table_top <= 0) {
254             return null;
255         }
256         _symbol_table_top--;
257         SymbolTable symbols = _symbol_table_stack[_symbol_table_top];
258         _symbol_table_stack[_symbol_table_top] = null;
259         return symbols;
260     }

274         IonReaderTextSpan(IonReaderTextUserX reader)

276             // TODO: convert _start_char_offset from a long and data page

347     private void hoistImpl(Span span)
348     {
349         if (!(span instanceof IonReaderTextSpan)) {
350             throw new IllegalArgumentException("position must match the reader");
351         }
352         IonReaderTextSpan text_span = (IonReaderTextSpan)span;
353 
354         UnifiedInputStreamX current_stream = _scanner.getSourceStream();
355         UnifiedDataPageX    curr_page      = text_span.getDataPage();
356         int                 array_offset   = (int)text_span._start_offset + _physical_start_offset;
357         int                 page_limit     = curr_page._page_limit;
358         int                 array_length   = page_limit - array_offset;
359 
360         // we're going to cast this value down.  Since we only support
361         // in memory single buffered chars here this is ok.
362         assert(text_span.getStartOffset() <= Integer.MAX_VALUE);
363 
364         // Now - create a new stream
365         // TODO: this is a pretty expensive way to do this. UnifiedInputStreamX
366         //       needs to have a reset method added that can reset the position
367         //       and length of the input to be some subset of the original source.
368         //       This would avoid a lot of object creation (and wasted destruction.
369         //       But this is a time-to-market solution here.  The change can be
370         //       made as support for streams is added.
371         UnifiedInputStreamX iis;
372         if (current_stream._is_byte_data) {
373             byte[] bytes = current_stream.getByteArray();
374             assert(bytes != null);
375             iis = UnifiedInputStreamX.makeStream(
376                                             bytes
377                                           , array_offset
378                                           , array_length
379                                       );
380         }
381         else {
382             char[] chars = current_stream.getCharArray();
383             assert(chars != null);
384             iis = UnifiedInputStreamX.makeStream(
385                                             chars
386                                           , array_offset
387                                           , array_length
388                                       );
389         }
390         IonType container = text_span.getContainerType();
391         re_init(iis, container, text_span._start_line, text_span._start_column);
392     }

398     @Override

400     {
401         if (facetType == SpanProvider.class)
402         {
403             return facetType.cast(new SpanProviderFacet());
404         }
405 
406         if (facetType == SeekableReader.class && _scanner.isBufferedInput())
407         {
408             return facetType.cast(new SeekableReaderFacet());
409         }
410
