17  package com.amazon.ion.impl;
18  
19  import static com.amazon.ion.TestUtils.GLOBAL_SKIP_LIST;
20  import static com.amazon.ion.TestUtils.GOOD_IONTESTS_FILES;
21  import static com.amazon.ion.TestUtils.testdataFiles;
22  
23  import com.amazon.ion.IonReader;
24  import com.amazon.ion.IonTestCase;
25  import com.amazon.ion.IonWriter;
26  import com.amazon.ion.junit.Injected.Inject;
27  import com.amazon.ion.system.IonTextWriterBuilder;
28  import com.amazon.ion.util.NullOutputStream;
29  import java.io.File;
30  import java.io.FileInputStream;
31  import java.io.InputStream;
32  import java.io.OutputStream;
33  import org.junit.Test;
34  
35  public class IonMarkupWriterFilesTest
36          extends IonTestCase
37  {
38      @Inject("testFile")
39      public static final File[] FILES = testdataFiles(GLOBAL_SKIP_LIST,
40                                               GOOD_IONTESTS_FILES);
41      private File               myTestFile;
42  
43      public void setTestFile(File file) {
44          myTestFile = file;
45      }
46  
47      /**
48       * Tests to make sure that the type, the output stream, and the
49       * annotation/field name aren't null for all the files. This will fail if
50       * the callback type isn't being set.
51       * {@link IonMarkupWriterTest#testStandardCallback()} should verify they're
52       * set correctly.
53       */
54      @Test
55      public void testPrintingStandard()
56          throws Exception
57      {
58          // Just ignore the output, since we're not checking it
59          OutputStream out = new NullOutputStream();
60          InputStream in = new FileInputStream(myTestFile);
61  
62          // Get a Test markup writer
63          _Private_IonTextWriterBuilder builder = (_Private_IonTextWriterBuilder)
64              IonTextWriterBuilder.standard();
65          IonWriter ionWriter = builder
66              .withCallbackBuilder(new TestMarkupCallback.Builder())
67              .build(out);
68  
69          // Load file into a reader
70          IonReader ionReader = system().newReader(in);
71  
72          assertNotNull("ionReader is null", ionReader);
73          assertNotNull("ionWriter is null", ionWriter);
74  
75          // Write data to the writer
76          ionWriter.writeValues(ionReader);
77          // Close the writer and reader
78          ionWriter.close();
79          ionReader.close();
80      }
81  }
