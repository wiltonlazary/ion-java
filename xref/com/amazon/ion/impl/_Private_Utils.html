285     public static SymbolToken localize(SymbolTable symtab,
286                                        SymbolToken sym)
287     {
288         String text = sym.getText();
289         int sid = sym.getSid();
290 
291         if (symtab != null)  // TODO amzn/ion-java/issues/21 require symtab
292         {
293             if (text == null)
294             {
295                 text = symtab.findKnownSymbol(sid);
296                 if (text != null)
297                 {
298                     sym = new SymbolTokenImpl(text, sid);
299                 }
300             }
301             else
302             {
303                 SymbolToken newSym = symtab.find(text);
304                 if (newSym != null)
305                 {
306                     sym = newSym;
307                 }
308                 else if (sid >= 0)
309                 {
310                     // We can't trust the sid, discard it.
311                     sym = new SymbolTokenImpl(text, UNKNOWN_SYMBOL_ID);
312                 }
313             }
314         }
315         else if (text != null && sid >= 0)
316         {
317             // We can't trust the sid, discard it.
318             sym = new SymbolTokenImpl(text, UNKNOWN_SYMBOL_ID);
319         }
320         return sym;
321     }

390     /**
391      * Encodes a String into bytes of a given encoding.
392      * <p>
393      * This method is preferred to {@link Charset#encode(String)} and
394      * {@link String#getBytes(String)} (<em>etc.</em>)
395      * since those methods will replace or ignore bad input, and here we throw
396      * an exception.
397      *
398      * @param s the string to encode.
399      *
400      * @return the encoded string, not null.
401      *
402      * @throws IonException if there's a {@link CharacterCodingException}.
403      */
404     public static byte[] encode(String s, Charset charset)
405     {
406         CharsetEncoder encoder = charset.newEncoder();
407         try
408         {
409             ByteBuffer buffer = encoder.encode(CharBuffer.wrap(s));
410             byte[] bytes = buffer.array();
411 
412             // Make another copy iff there's garbage after the limit.
413             int limit = buffer.limit();
414             if (limit < bytes.length)
415             {
416                 bytes = copyOf(bytes, limit);
417             }
418 
419             return bytes;
420         }
421         catch (CharacterCodingException e)
422         {
423             throw new IonException("Invalid string data", e);
424         }
425     }

428     /**
429      * Decodes a byte sequence into a string, given a {@link Charset}.
430      * <p>
431      * This method is preferred to {@link Charset#decode(ByteBuffer)} and
432      * {@link String#String(byte[], Charset)} (<em>etc.</em>)
433      * since those methods will replace or ignore bad input, and here we throw
434      * an exception.
435      *
436      * @param bytes the data to decode.
437      *
438      * @return the decoded string, not null.
439      *
440      * @throws IonException if there's a {@link CharacterCodingException}.
441      */
442     public static String decode(byte[] bytes, Charset charset)
443     {
444         CharsetDecoder decoder = charset.newDecoder();
445         try
446         {
447             CharBuffer buffer = decoder.decode(ByteBuffer.wrap(bytes));
448             return buffer.toString();
449         }
450         catch (CharacterCodingException e)
451         {
452             String message =
453                 "Input is not valid " + charset.displayName() + " data";
454             throw new IonException(message, e);
455         }
456     }

503     public static byte[] convertUtf16UnitsToUtf8(String text)

542     /**
543      * Calls {@link InputStream#read(byte[], int, int)} until the requested
544      * length is read or EOF is encountered.
545      * This method will block until the request is satisfied.
546      *
547      * @param in        The stream to read from.
548      * @param buf       The buffer to read to.
549      * @param offset    The offset of the buffer to read from.
550      * @param length    The length of the data to read.
551      *
552      * @return the number of bytes read from the stream.  May be less than
553      *  {@code length} if EOF is encountered before reading that far.
554      *
555      * @see #readFully(InputStream, byte[])
556      */
557     public static int readFully(InputStream in, byte[] buf,
558                                 int offset, int length)
559     throws IOException
560     {
561         int readBytes = 0;
562         while (readBytes < length)
563         {
564             int amount = in.read(buf, offset, length - readBytes);
565             if (amount < 0)
566             {
567                 // EOF
568                 return readBytes;
569             }
570             readBytes += amount;
571             offset += amount;
572         }
573         return readBytes;
574     }

577     public static byte[] loadFileBytes(File file)
578         throws IOException
579     {
580         long len = file.length();
581         if (len < 0 || len > Integer.MAX_VALUE) {
582             throw new IllegalArgumentException("File too long: " + file);
583         }
584 
585         byte[] buf = new byte[(int) len];
586 
587         FileInputStream in = new FileInputStream(file);
588         try {
589             int readBytesCount = in.read(buf);
590             if (readBytesCount != len || in.read() != -1)
591             {
592                 throw new IOException("Read the wrong number of bytes from "
593                                        + file);
594             }
595         }
596         finally {
597             in.close();
598         }
599 
600         return buf;
601     }

640     public static boolean streamIsIonBinary(PushbackInputStream pushback)
641         throws IonException, IOException
642     {
643         boolean isBinary = false;
644         byte[] cookie = new byte[_Private_IonConstants.BINARY_VERSION_MARKER_SIZE];

646         int len = readFully(pushback, cookie);
647         if (len == _Private_IonConstants.BINARY_VERSION_MARKER_SIZE) {

649         }
650         if (len > 0) {
651             pushback.unread(cookie, 0, len);
652         }
653         return isBinary;

693     public static boolean symtabIsLocalAndNonTrivial(SymbolTable symtab)
694     {
695         if (symtab == null) return false;
696         if (!symtab.isLocalTable()) return false;
697 
698         // If symtab has imports we must retain it.
699         // Note that I chose to retain imports even in the degenerate case
700         // where the imports have no symbols.
701         if (symtab.getImportedTables().length > 0) {
702             return true;
703         }
704 
705         if (symtab.getImportedMaxId() < symtab.getMaxId()) {
706             return true;
707         }
708 
709         return false;
710     }

713     /**
714      * Is the table null, system, or local without imported symbols?
715      */
716     public static boolean isTrivialTable(SymbolTable table)
717     {
718         if (table == null)         return true;
719         if (table.isSystemTable()) return true;
720         if (table.isLocalTable()) {
721             // this is only true when there are no local
722             // symbols defined
723             // and there are no imports with any symbols
724             if (table.getMaxId() == table.getSystemSymbolTable().getMaxId()) {
725                 return true;
726             }
727         }
728         return false;
729     }

879     /**
880      * Determines, for two local symbol tables, whether the passed-in {@code superset} symtab is an extension
881      * of {@code subset}.  This works independent of implementation details--particularly in cases
882      * where {@link LocalSymbolTable#symtabExtends(SymbolTable)} cannot be used.
883      *
884      * @see #symtabExtends(SymbolTable, SymbolTable)
885      */
886     private static boolean localSymtabExtends(SymbolTable/../../com/amazon/ion/SymbolTable.html#SymbolTable">SymbolTable superset, SymbolTable subset)
887     {
888         if (subset.getMaxId() > superset.getMaxId())
889         {
890             // the subset has more symbols
891             return false;
892         }
893 
894         // NB this API almost certainly requires cloning--symbol table's API doesn't give us a way to polymorphically
895         //    get this without materializing an array
896         final SymbolTable[] supersetImports     = superset.getImportedTables();
897         final SymbolTable[] subsetImports       = subset.getImportedTables();
898 
899         // TODO this is over-strict, but not as strict as LocalSymbolTable.symtabExtends()
900         if (supersetImports.length != subsetImports.length)
901         {
902             return false;
903         }
904         // NB we cannot trust Arrays.equals--we don't know how an implementation will implement it...
905         for (int i = 0; i < supersetImports.length; i++)
906         {
907             final SymbolTable supersetImport = supersetImports[i];
908             final SymbolTable subsetImport = subsetImports[i];
909             if (!supersetImport.getName().equals(subsetImport.getName())
910                  || supersetImport.getVersion() != subsetImport.getVersion())
911             {
912                 // bad match on import
913                 return false;
914             }
915         }
916 
917         // all the imports lined up, lets make sure the locals line up too
918         final Iterator<String> supersetIter     = superset.iterateDeclaredSymbolNames();
919         final Iterator<String> subsetIter       = subset.iterateDeclaredSymbolNames();
920         while (subsetIter.hasNext())
921         {
922             final String nextSubsetSymbol       = subsetIter.next();
923             final String nextSupersetSymbol     = supersetIter.next();
924             if (!nextSubsetSymbol.equals(nextSupersetSymbol))
925             {
926                 // local symbol mismatch
927                 return false;
928             }
929         }
930 
931         // we made it this far--superset is really a superset of subset
932         return true;
933     }

937      * of {@code subset}.

951     {
952         assert superset.isSystemTable() || superset.isLocalTable();
953         assert subset.isSystemTable() || subset.isLocalTable();
954 
955         // NB: system symtab 1.0 is a singleton, hence if both symtabs
956         //     are one this will be true.
957         if (superset == subset) return true;
958 
959         // If the subset's symtab is a system symtab, the superset's is always
960         // an extension of the subset's as system symtab-ness is irrelevant to
961         // the conditions for copy opt. to be safe.
962         // TODO amzn/ion-java/issues/24 System symtab-ness ARE relevant if there's multiple
963         //      versions.
964         if (subset.isSystemTable()) return true;
965 
966         // From here on, subset is a LST because isSystemTable() is false.
967 
968         if (superset.isLocalTable())

983         // superset extends subset
984         return subset.getMaxId() == superset.getMaxId();
