15  package software.amazon.ion;
16  import java.math.BigDecimal;
17  
18  
19  /**
20   * An Ion <code>float</code> value.
21   * <p>
22   * <b>WARNING:</b> This interface should not be implemented or extended by
23   * code outside of this library.
24   *
25   * <h2>Precision Problems</h2>
26   *
27   * Use of binary floating-point numbers is prone to countless problems.
28   * The vast majority of applications should use {@code decimal} values instead.
29   * Please read the
30   * <a href="http://speleotrove.com/decimal/decifaq1.html">Decimal Arithmetic
31   * FAQ</a> for horror stories.
32   * <p>
33   * If you have any doubt whatsoever on whether you should use {@code float} or
34   * {@code decimal}, then you should use {@code decimal}.
35   *
36   * @see IonDecimal
37   */
38  public interface IonFloat
39      extends IonValue
40  {
41      /**
42       * Gets the value of this Ion <code>float</code> as a Java
43       * <code>float</code> value.
44       *
45       * @return the float value.
46       * @throws NullValueException if <code>this.isNullValue()</code>.
47       */
48      public float floatValue()
49          throws NullValueException;
50  
51      /**
52       * Gets the value of this Ion <code>float</code> as a Java
53       * <code>double</code> value.
54       *
55       * @return the double value.
56       * @throws NullValueException if <code>this.isNullValue()</code>.
57       */
58      public double doubleValue()
59          throws NullValueException;
60  
61  
62      /**
63       * Gets the value of this Ion {@code float} as a Java {@link BigDecimal}.
64       * This follows the behavior of {@link BigDecimal#valueOf(double)}. It's
65       * recommended to call {@link IonFloat#isNumericValue()} before calling
66       * this method.
67       *
68       * @return the {@link BigDecimal} value, or {@code null} if
69       * {@code this.isNullValue()}.
70       *
71       * @throws NumberFormatException if this value is {@code nan},
72       * {@code +inf}, or {@code -inf}, because {@link BigDecimal} cannot
73       * represent those values.
74       */
75      public BigDecimal bigDecimalValue()
76          throws NullValueException;
77  
78      /**
79       * Sets the value of this element.
80       */
81      public void setValue(float value);
82  
83      /**
84       * Sets the value of this element.
85       */
86      public void setValue(double value);
87  
88      /**
89       * Sets the value of this element.
90       * Since Ion {@code float}s are essentially Java {@code double}s,
91       * this performs a narrowing conversion as described by
92       * {@link BigDecimal#doubleValue()}.
93       *
94       * @param value the new value of this float;
95       * may be <code>null</code> to make this <code>null.float</code>.
96       */
97      public void setValue(BigDecimal value);
98  
99      /**
100      * Determines whether this value is numeric. Returns true if this value
101      * is none of {@code null}, {@code nan}, {@code +inf}, and {@code -inf},
102      * and false if it is any of them.
103      *
104      * @return a checked condition whether this value is numeric.
105      */
106     public boolean isNumericValue();
107 
108     public IonFloat clone()
109         throws UnknownSymbolException;
110 }
