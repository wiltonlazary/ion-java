42  /**
43   * Provides equivalence comparisons between two {@link IonValue}s, following
44   * the contract of {@link IonValue#equals(Object)}.
45   *
46   * <p>
47   * Basic usage of this class is as follows:
48   *
49   *<pre>
50   *    IonValue v1 = ...;
51   *    IonValue v2 = ...;
52   *    com.amazon.ion.util.Equivalence.ionEquals( v1, v2 );
53   *</pre>
54   *
55   * More likely, a static import would make using this class easier.
56   *
57   *<pre>
58   *    import static com.amazon.ion.util.Equivalence.ionEquals;
59   *    ...
60   *    boolean equivalent = ionEquals( v1, v2 );
61   *</pre>
62   *
63   * <p>
64   * <h3>Ion Equivalence</h3>
65   * In order to make Ion a useful model to describe data, we must first define
66   * the notion of equivalence for all values representable in Ion. Equivalence
67   * with respect to Ion values means that if two Ion values, X and Y, are
68   * equivalent, they represent the same data and can be substituted for the other
69   * without loss of information.
70   *
71   * This relationship is:
72   * <ul>
73   *   <li>
74   *     symmetric: X is equivalent to Y if and only if Y is equivalent to X.
75   *   </li>
76   *   <li>
77   *     transitive: if X is equivalent to Y and Y is equivalent to Z, then X is
78   *     equivalent to Z.
79   *   </li>
80   *   <li>
81   *     reflexive: X is equivalent to X.
82   *   </li>
83   * </ul>
84   *
85   * <h4>Ordered Sequence Equivalence</h4>
86   * When an ordered sequence (i.e. tuple) of elements is specified in this
87   * document, equivalence over such an ordered sequence is defined as follows.
88   *
89   * A tuple, A = (a1, a2, ..., an), is equivalent to another tuple, B = (b1, b2,
90   * ..., bm) if and only if the cardinality (number of elements) of A equals the
91   * cardinality of B (i.e. n == m) and ai is equivalent to bi for i = 1 ... n.
92   *
93   * <h4>Un-Ordered Sequence Equivalence</h4>
94   * When an un-ordered sequence (i.e. bag or multi-set) is specified in this
95   * document, equivalence over such a sequence is defined as follows.
96   *
97   * A bag, A = {a1, a2, ..., an} is equivalent to B = {b1, b2, ..., bm} if and
98   * only if the cardinality of A is equal to the cardinality of B and for each
99   * element, x, in A there exists a distinct element, y, in B for which x is
100  * equivalent to y.
101  *
102  * <h4>Values</h4>
103  * Any arbitrary, atomic value in the Ion Language can be denoted as the tuple,
104  * (A, V), where A is an ordered list of annotations, and V is an Ion Primitive
105  * Data or Ion Complex Data value. The list of annotations, A is an tuple of Ion
106  * Symbols (a specific type of Ion Primitive).
107 
108  */
109 public final class Equivalence {

122     private static int compareAnnotations(SymbolToken[] ann1,
123                                           SymbolToken[] ann2)
124     {
125         int len = ann1.length;
126         int result = len - ann2.length;
127 
128         if (result == 0) {
129             for (int i=0; (result == 0) && (i < len); i++) {
130                 result = compareSymbolTokens(ann1[i], ann2[i]);
131             }
132         }
133 
134         return result;
135     }

138     private static int compareSymbolTokens(SymbolToken tok1,
139                                            SymbolToken tok2)
140     {
141         String text1 = tok1.getText();
142         String text2 = tok2.getText();
143         if (text1 == null || text2 == null) {
144             if (text1 != null) return  1;
145             if (text2 != null) return -1;
146 
147             int sid1 = tok1.getSid();
148             int sid2 = tok2.getSid();
149             if (sid1 < sid2) return -1;
150             if (sid1 > sid2) return  1;
151             return 0;
152         }
153 
154         return text1.compareTo(text2);
155     }

158     /**
159      * Converts an IonStruct to a multi-set for use in IonStruct equality
160      * checks. This method returns the multi-set as a {@code Map<Field, Field>}.
161      * <p>
162      * A multi-set supports order-independent equality, and may have duplicate
163      * elements.
164      * <p>
165      * Take special note that {@link Set} is missing a {@code get()} API,
166      * and cannot contain duplicate elements, hence we cannot use it.
167      */
168     private static final Map<Field, Field>
169         convertToMultiSet(final IonStruct struct, final boolean strict) {
170 
171         final Map<Field, Field> structMultiSet =
172             new HashMap<Field, Field>();
173 
174         for (final IonValue val : struct) {
175             final Field item = new Field(val, strict);
176             Field curr = structMultiSet.put(item, item);
177             // curr will be non-null if the multi-set already contains the
178             // name/value pair
179             if (curr != null) {
180                 // Set the 'occurrences' of the Field that is contained in
181                 // the multi-set (i.e. item) to that of the previously mapped
182                 // Field (i.e. curr)
183                 item.occurrences = curr.occurrences;
184             }
185             // At this point, item will be an existing
186             // name/value pair in the multi-set - increment its occurrence
187             item.occurrences++;
188         }
189 
190         return structMultiSet;
191     }

194     private static int compareStructs(final IonStruct s1,
195                                       final IonStruct s2,
196                                       boolean strict)
197     {
198         int result = s1.size() - s2.size();
199         if (result == 0) {
200             // We convert IonStruct s1 to a multi-set (which is a
201             // Map<Field, Field>). Refer to convertToMultiSet()'s
202             // documentation for more info
203             final Map<Field, Field> s1MultiSet
204                     = convertToMultiSet(s1, strict);
205 
206             // Iterates through each name/value pair in IonStruct s2 and
207             // determine if it also occurs in s1MultiSet.
208             // During each iteration:
209             //          If it does, remove an occurrence from s1MultiSet
210             //          If it doesn't, the two IonStructs aren't equal
211             for (IonValue val : s2) {
212                 Field field = new Field(val, strict);
213 
214                 // Find an occurrence of the name/value pair in s1MultiSet
215                 Field mappedValue = s1MultiSet.get(field);
216 
217                 if (mappedValue == null || mappedValue.occurrences == 0) {
218                     // No match in occurrences, the IonStructs aren't equal
219                     return -1;
220                 }
221 
222                 // Remove an occurrence by decrementing count instead of
223                 // explicitly calling Map.remove(), as Map.remove() is a slower
224                 // operation
225                 mappedValue.occurrences--;
226             }
227         }
228         return result;
229     }

232     private static int compareSequences(final IonSequence s1,
233                                         final IonSequence s2,
234                                         boolean strict)
235     {
236         int result = s1.size() - s2.size();
237         if (result == 0) {
238             Iterator<IonValue> iter1 = s1.iterator();
239             Iterator<IonValue> iter2 = s2.iterator();
240             while (iter1.hasNext()) {
241                 result = ionCompareToImpl(iter1.next(),
242                                           iter2.next(),
243                                           strict);
244                 if (result != 0) {
245                     break;
246                 }
247             }
248         }
249         return result;
250     }

253     /** Compare LOB content by stream--assuming non-null. */
254     private static int compareLobContents(final IonLobIonLob">IonLob lob1, final IonLob lob2)
255     {
256         int in1 = lob1.byteSize();
257         int in2 = lob2.byteSize();
258         int result = (in1- in2);
259 
260         if (result == 0) {
261             final InputStream stream1 = lob1.newInputStream();
262             final InputStream stream2 = lob2.newInputStream();
263 
264             // too bad Java doesn't do RAII with better syntax...
265             try {
266               try {
267                 try {
268                   while (result == 0) {
269                       in1 = stream1.read();
270                       in2 = stream2.read();
271                       if (in1 == -1 || in2 == -1) {
272                         if (in1 != -1) result = 1;
273                         if (in2 != -1) result = -1;
274                         break;
275                       }
276                       result = (in1 - in2);
277                   }
278                 } finally {
279                   stream1.close();
280                 }
281               } finally {
282                 stream2.close();
283               }
284             } catch (final IOException e) {
285                 // XXX hopefully won't happen with LOB streams--if it does rethrow unchecked.
286                 //     this would violate Object.equals() would it not?
287                 throw new IonException(e);
288             }
289         }
290 
291         return result;
292     }

295     /**
296      * Class that denotes a name/value pair in Structs.
297      * <p>
298      * Structs are unordered collections of name/value pairs. The names are
299      * symbol tokens, and the values are unrestricted. Each name/value pair is
300      * called a <em>field</em>.
301      * <p>
302      * The responsibilities of this class is to expose a
303      * {@code Map<Field, Field>} as a multi-set. A Field instance holds the name
304      * and value of a Struct field, and counts the number of times that
305      * name/value pair occurs within the multi-set.
306      * <p>
307      * For example, an IonStruct:
308      *<pre>
309      *  {
310      *    a : 123,
311      *    a : 123
312      *  }
313      *</pre>
314      * will be converted into a multi-set {@code Map<Field, Field>} with
315      * a single {@code Field} -> {@code Field} with {@code occurrences} of 2.
316      * <p>
317      * Refer to
318      * {@link Equivalence#convertToMultiSet(IonStruct, boolean)} and
319      * {@link Field#equals(Object)} for more info.
320      * <p>
321      * NOTE: This class should only be instantiated for the sole purpose of
322      * using it as either a <em>key</em> or <em>value</em> in a {@link Map}.
323      */
324     static class Field {
325         private final String    name; // aka field name
326         private final IonValue  value;
327         private final boolean   strict;
328 
329         /**
330          * Number of times that this specific field (with the same name
331          * and value) occurs within the struct.
332          */
333         private int occurrences;
334 
335         Field(final IonValue value, final boolean strict)
336         {
337             SymbolToken tok = value.getFieldNameSymbol();
338             String name = tok.getText();
339             if (name == null) {
340                 // TODO amzn/ion-java/issues/23 Problematic with unknown field names.
341                 name = UNKNOWN_SYMBOL_TEXT_PREFIX + tok.getSid();
342             }
343             this.name = name;
344             this.value = value;
345             this.strict = strict;
346 
347             // Occurrences of this name/value pair is 0 initially
348             this.occurrences = 0;
349         }
350 
351         @Override
352         public int hashCode() {
353             return name.hashCode();
354             // TODO amzn/ion-java/issues/58 : implement hash code such that it respects
355             // 'strict'. The prevously attempted fix is commented out below but
356             // is not sufficient because value.hasCode will always include
357             // type annotations in the hash computation. Type annotations
358             // should not be part of the hash computation if strict=true.
359 //            result = (31 * result) + value.hashCode();
360 //            return result;
361         }
362 
363         /**
364          * This method is implicitly called by {@link Map#get(Object)} to
365          * obtain a value to which {@code other} is mapped.
366          */
367         @Override
368         public boolean equals(final Object other) {
369             // We can assume other is always a Field and strict
370             // is the same - internal usage dictates it.
371             final Field sOther = (Field) other;
372 
373             return name.equals(sOther.name)
374                 && ionEqualsImpl(value, ((Field) other).value, strict);
375         }
376     }

378     private static boolean ionEqualsImpl(final IonValue v1,
379                                                final IonValue v2,
380                                                final boolean strict)
381     {
382         return (ionCompareToImpl(v1, v2, strict) == 0);
383     }

385     private static int ionCompareToImpl(final IonValue v1,
386                                         final IonValue v2,
387                                         final boolean strict)
388     {
389         int result = 0;
390 
391         if (v1 == null || v2 == null) {
392             if (v1 != null) result = 1;
393             if (v2 != null) result = -1;
394             // otherwise v1 == v2 == null and result == 0
395             return result;
396         }
397 
398         // check type
399         IonType ty1 = v1.getType();
400         IonType ty2 = v2.getType();
401         result = ty1.compareTo(ty2);
402 
403         if (result == 0) {
404             boolean bo1 = v1.isNullValue();
405             boolean bo2 = v2.isNullValue();
406 
407             if (bo1 || bo2) {
408                  // null combination case--we already know they are
409                  // the same type
410                 if (!bo1) result = 1;
411                 if (!bo2) result = -1;
412                 // othersize they're equal (and null values)
413             }
414             else {
415                 // value compare only if both are not null
416                 switch (ty1)
417                 {
418                 case NULL:
419                     // never visited, precondition is that both are not null
420                     break;
421                 case BOOL:
422                     if (((IonBool) v1).booleanValue()) {
423                         result = ((IonBool) v2).booleanValue() ? 0 : 1;
424                     }
425                     else {
426                         result = ((IonBool) v2).booleanValue() ? -1 : 0;
427                     }
428                     break;
429                 case INT:
430                     result = ((IonInt) v1).bigIntegerValue().compareTo(
431                              ((IonInt) v2).bigIntegerValue());
432                     break;
433                 case FLOAT:
434                     result = Double.compare(((IonFloat) v1).doubleValue(),
435                                             ((IonFloat) v2).doubleValue());
436                     break;
437                 case DECIMAL:
438                     assert !PUBLIC_COMPARISON_API; // TODO amzn/ion-java/issues/26
439                     result = Decimal.equals(((IonDecimal) v1).decimalValue(),
440                                             ((IonDecimal) v2).decimalValue())
441                                             ? 0 : 1;
442                     break;
443                 case TIMESTAMP:
444                     if (strict) {
445                         assert !PUBLIC_COMPARISON_API; // TODO amzn/ion-java/issues/26
446                         result = (((IonTimestamp) v1).timestampValue().equals(
447                                   ((IonTimestamp) v2).timestampValue())
448                                   ? 0 : 1);
449                     }
450                     else {
451                         // This is kind of lying here, the 'strict' boolean
452                         // (if false) denotes ONLY that annotations are not
453                         // check for equality. But what this is doing here is
454                         // that it is also ignoring IonTimesamps' precision and
455                         // local offset.
456                         result = ((IonTimestamp) v1).timestampValue().compareTo(
457                                  ((IonTimestamp) v2).timestampValue());
458                     }
459                     break;
460                 case STRING:
461                     result = (((IonText) v1).stringValue()).compareTo(
462                               ((IonText) v2).stringValue());
463                     break;
464                 case SYMBOL:
465                     result = compareSymbolTokens(((IonSymbol) v1).symbolValue(),
466                                                  ((IonSymbol) v2).symbolValue());
467                     break;
468                 case BLOB:
469                 case CLOB:
470                     result = compareLobContents((IonLobef="../../../../com/amazon/ion/IonLob.html#IonLob">IonLob) v1, (IonLob) v2);
471                     break;
472                 case STRUCT:
473                     assert !PUBLIC_COMPARISON_API; // TODO amzn/ion-java/issues/26
474                     result = compareStructs((IonStruct) v1,
475                                             (IonStruct) v2,
476                                             strict);
477                     break;
478                 case LIST:
479                 case SEXP:
480                 case DATAGRAM:
481                     result = compareSequences((IonSequence) v1,
482                                               (IonSequence) v2,
483                                               strict);
484                     break;
485                 }
486             }
487         }
488 
489         // if the values are otherwise equal, but the caller wants strict
490         // comparison, then we check the annotations
491         if ((result == 0) && strict) {
492             // check tuple equality over the annotations
493             // (which are symbol tokens)
494             result = compareAnnotations(v1.getTypeAnnotationSymbols(),
495                                         v2.getTypeAnnotationSymbols());
496         }
497 
498         return result;
499     }
