229     /**
230      * Constructs an Ion-encoded string literal with a single escaped character.
231      * For example, <code>makeEscapedCharString('n')</code> returns
232      * (Java) <code>"\"\\n\""</code>, equivalent to Ion <code>"\n"</code>.
233      */
234     protected static String makeEscapedCharString(char escape)
235     {
236         final String QT = "\"";
237         final String BS = "\\";
238 
239         String result = QT + BS + escape + QT;
240         return result;
241     }

255     public IonDatagram reload(IonDatagram dg)
256     {
257         byte[] bytes = dg.getBytes();
258         checkBinaryHeader(bytes);
259 
260         IonDatagram dg1 = loader().load(bytes);
261         return dg1;
262     }

264     /**
265      * Put value into a datagram, get bytes, and reload to single value.
266      * Result should be equivalent.
267      */
268     public IonValue reload(IonValue value)
269     {
270         IonDatagram dg = system().newDatagram(value);
271         byte[] bytes = dg.getBytes();
272         checkBinaryHeader(bytes);
273 
274         try {
275             dg = loader().load(bytes);
276         } catch (final IonException e) {
277             final String hex = BinaryTest.bytesToHex(bytes);
278             throw new IonException("Bad bytes: " + hex, e);
279         }
280         assertEquals(1, dg.size());
281         return dg.get(0);
282     }

331     /**
332      * Parses text as a single Ion value.  If the text contains more than that,
333      * a failure is thrown.
334      *
335      * @param iterator must not be <code>null</code>.
336      * @return a single value, or <code>null</code> if the scanner has nothing
337      * on it.
338      */
339     public IonValue singleValue(Iterator<IonValue> iterator)
340     {
341         IonValue value = null;
342 
343         if (iterator.hasNext())
344         {
345             value = iterator.next();
346 
347             if (iterator.hasNext())
348             {
349                 IonValue part = iterator.next();
350                 fail("Found unexpected part: " + part);
351             }
352         }
353 
354         return value;
355     }

359     /**
360      * Parses text as a single Ion value.  If the text contains more than that,
361      * a failure is thrown.
362      *
363      * @param text must not be <code>null</code>.
364      * @return a single value, not <code>null</code>.
365      */
366     public IonValue oneValue(String text)
367     {
368         IonValue value = null;
369 
370         Iterator<IonValue> iterator = system().iterate(text);
371         if (iterator.hasNext())
372         {
373             value = iterator.next();
374 
375             if (iterator.hasNext())
376             {
377                 IonValue part = iterator.next();
378                 fail("Found unexpected part <" + part + "> in text: " + text);
379             }
380         }
381         else
382         {
383             fail("No data found in text: " + text);
384         }
385 
386         return value;
387     }

459     public static void checkBinaryHeader(byte[] datagram)

466             assertEquals("datagram cookie byte 2", 0x01, datagram[1] & 0xff);

705      * Checks that the value is an IonSymbol with the given name.
706      */
707     public static void checkSymbol(String name, int id, IonValue value)
708     {
709         checkType(IonType.SYMBOL, value);
710         IonSymbol sym = (IonSymbol) value;
711 
712         assertFalse(value.isNullValue());
713 
714         if (name == null)
715         {
716             try {
717                 sym.stringValue();
718                 fail("Expected " + UnknownSymbolException.class);
719             }
720             catch (UnknownSymbolException e)
721             {
722                 assertEquals(id, e.getSid());
723             }
724         }
725         else
726         {
727             assertEquals("symbol name", name, sym.stringValue());
728         }
729 
730         // just so we can set a break point on this before we lose all context
731         int sid = sym.symbolValue().getSid();
732         if (sid != id) {
733             assertEquals("symbol id", id, sym.symbolValue().getSid());
734         }
735 
736         checkSymbol(name, id, sym.symbolValue());
737     }

745     public static void checkSymbol(String text, int sid, boolean dupe,
746                                    SymbolTable symtab)
747     {
748         assert !dupe || text != null;
749 
750         String msg = "text:" + text + " sid:" + sid;
751 
752         if (text != null)
753         {
754             if (sid != UNKNOWN_SYMBOL_ID)
755             {
756                 assertEquals(msg, text, symtab.findKnownSymbol(sid));
757             }
758 
759             // Can't do this stuff when we have a duplicate symbol.
760             if (! dupe)
761             {
762                 assertEquals(msg, sid, symtab.findSymbol(text));
763 
764                 SymbolToken symToken = symtab.intern(text);
765                 assertEquals(msg, sid, symToken.getSid());
766 
767                 symToken = symtab.find(text);
768                 assertEquals(msg, sid, symToken.getSid());
769             }
770         }
771         else // No text expected, must have sid
772         {
773             assertEquals(msg, text /* null */, symtab.findKnownSymbol(sid));
774         }
775     }

834         assertTrue("table isn't local", symtab.isLocalTable());
835         assertFalse("table is shared",  symtab.isSharedTable());
836         assertFalse("table is system",  symtab.isSystemTable());
837         assertFalse("table is substitute", symtab.isSubstitute());
838         assertNotNull("table has imports", symtab.getImportedTables());
839 
840         checkUnknownSymbol(" not defined ", UNKNOWN_SYMBOL_ID, symtab);
841 
842         SymbolTable system = symtab.getSystemSymbolTable();
843         checkSystemTable(system);
844         assertEquals(system.getIonVersionId(), symtab.getIonVersionId());
845     }

877     public static SymbolTable findImportedTable(SymbolTable localTable,
878                                                 String importName)
879     {
880         SymbolTable[] imports = localTable.getImportedTables();
881         if (imports == null) return null;
882 
883         for (int i = 0; i < imports.length; i++)
884         {
885             SymbolTable current = imports[i];
886             // FIXME why does this allow null?
887             if (current != null && importName.equals(current.getName()))
888             {
889                 return current;
890             }
891         }
892 
893         return null;
894     }
