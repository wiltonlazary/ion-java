23  final class IonStringLite
24      extends IonTextLite
25      implements IonString
26  {
27      private static final int HASH_SIGNATURE =
28          IonType.STRING.toString().hashCode();
29  
30      /**
31       * Constructs a <code>null.string</code> value.
32       */
33      IonStringLite(ContainerlessContext context, boolean isNull)
34      {
35          super(context, isNull);
36      }
37

39      {
40          super(existing, context);
41          // no need to set values as these are set at the parent IonTextLite
42      }
43  
44      @Override
45      IonStringLite clone(IonContext parentContext)
46      {
47          return new IonStringLite(this, parentContext);
48      }
49  
50      @Override
51      public IonStringLite clone()
52      {
53          return clone(ContainerlessContext.wrap(getSystem()));
54      }
55  
56      @Override
57      int hashCode(SymbolTableProvider symbolTableProvider)
58      {
59          int result = HASH_SIGNATURE;
60  
61          if (!isNullValue()) {
62              result ^= stringValue().hashCode();
63          }
64  
65          return hashTypeAnnotations(result, symbolTableProvider);
66      }
67  
68      @Override
69      public IonType getType()
70      {
71          return IonType.STRING;
72      }
73  
74      @Override
75      final void writeBodyTo(IonWriter writer, SymbolTableProvider symbolTableProvider)
76          throws IOException
77      {
78          writer.writeString(_get_value());
79      }
80  
81      @Override
82      public void accept(ValueVisitor visitor) throws Exception
83      {
84          visitor.visit(this);
85      }

87  }
