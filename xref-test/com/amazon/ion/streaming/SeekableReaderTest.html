42      private void checkSpans(IonDatagram dg, Span[] positions)
43      {
44          for (int i = dg.size() - 1; i >= 0; i--)
45          {
46              hoist(positions[i]);
47              IonType dg_type = dg.get(i).getType();
48              IonType span_type = in.next();
49              if (dg_type.equals(span_type) == false) {
50                  assertEquals(dg_type, span_type);
51              }
52              expectTopLevel();
53              IonAssert.assertIonEquals(dg.get(i), system().newValue(in));
54          }
55          expectTopEof();
56      }

73      @Test
74      public void testWalkingBackwards()
75      {
76          String text =
77              "null true 3 4e0 5.0 6666-06-06T '7' \"8\" {{\"\"}} {{}} [] () {}";
78  
79          IonDatagram dg = loader().load(text);
80  
81          Span[] positions = new Span[dg.size()];
82  
83          read(text);
84          for (int i = 0; i < dg.size(); i++)
85          {
86              assertEquals(dg.get(i).getType(), in.next());
87              positions[i] = sr.currentSpan();
88          }
89          expectEof();
90  
91          checkSpans(dg, positions);
92  
93  
94          // Collect spans *after* extracting scalar body.
95  
96          read(text);
97          for (int i = 0; i < dg.size(); i++)
98          {
99              IonType t =  in.next();
100             assertEquals(dg.get(i).getType(),t);
101             if (! IonType.isContainer(t))
102             {
103                 TestUtils.consumeCurrentValue(in);
104             }
105             positions[i] = sr.currentSpan();
106         }
107         expectTopEof();
108 
109         checkSpans(dg, positions);
110     }

113     @Test
114     public void testHoistingWithinContainers()
115     {
116         read("{f:v,g:[c, (d), e], /* h */ $0:null} s");
117 
118         in.next();
119         in.stepIn();
120             in.next();
121             Span fPos = sr.currentSpan();
122             assertEquals("v", in.stringValue());
123             in.next();
124             Span gPos = sr.currentSpan();
125             in.stepIn();
126                 in.next();
127                 assertEquals("c", in.stringValue());
128                 Span cPos = sr.currentSpan();
129             in.stepOut();
130             in.next();
131             Span hPos = sr.currentSpan();
132             expectEof();
133         in.stepOut();
134         in.next();
135         Span sPos = sr.currentSpan();
136         expectTopEof();
137 
138 
139         hoist(fPos);
140         IonType in_type = in.next();
141         assertEquals(IonType.SYMBOL, in_type);
142         expectTopLevel();
143         assertEquals("v", in.stringValue());
144         expectTopEof();
145 
146         hoist(cPos);
147         in.next();
148         expectTopLevel();
149         assertEquals("c", in.stringValue());
150         expectTopEof();
151 
152         hoist(gPos);
153         assertEquals(IonType.LIST, in.next());
154         expectTopLevel();
155         in.stepIn();
156             in.next();
157             assertEquals("c", in.stringValue());
158             assertEquals(IonType.SEXP, in.next()); // (d)
159             in.stepIn();
160                 in.next();
161                 assertEquals("d", in.stringValue());
162                 expectEof();
163             in.stepOut();
164             assertEquals(IonType.SYMBOL, in.next());
165             assertEquals("e", in.stringValue());
166             expectEof();
167         in.stepOut();
168         expectTopEof();
169 
170         hoist(hPos);
171         assertEquals(IonType.NULL, in.next());
172         expectTopLevel();
173 
174         hoist(fPos);
175         check().noFieldName();
176         assertEquals(IonType.SYMBOL, in.next());
177         expectTopLevel();
178         assertEquals("v", in.stringValue());
179         expectTopEof();
180 
181         hoist(sPos);
182         assertEquals(IonType.SYMBOL, in.next());
183         expectTopLevel();
184         assertEquals("s", in.stringValue());
185         expectTopEof();
186     }
