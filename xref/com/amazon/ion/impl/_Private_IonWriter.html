16  package com.amazon.ion.impl;
17  
18  import com.amazon.ion.IonCatalog;
19  import com.amazon.ion.IonWriter;
20  import java.io.IOException;
21  
22  /**
23   * NOT FOR APPLICATION USE!
24   */
25  public interface _Private_IonWriter
26      extends IonWriter
27  {
28      /** Mostly for testing at this point, but could be useful public API. */
29      IonCatalog getCatalog();
30  
31      /**
32       * Returns true if the field name has been set either through setFieldName or
33       * setFieldId.  This is generally more efficient than calling getFieldName or
34       * getFieldId and checking the return type as it does not need to resolve the
35       * name through a symbol table.  This returns false if the field name has not
36       * been set.
37       *
38       * @return true if a field name has been set false otherwise
39       */
40      boolean isFieldNameSet();
41  
42      /**
43       * Returns the current depth of containers the writer is at.  This is
44       * 0 if the writer is at top-level.
45       * @return int depth of container nesting
46       */
47      int getDepth();
48  
49      /**
50       * Write an Ion version marker symbol to the output.  This
51       * is the $ion_1_0 value currently (in later versions the
52       * number may change).  In text output this appears as the
53       * text symbol.  In binary this will be the symbol id if
54       * the writer is in a list, sexp or struct.  If the writer
55       * is currently at the top level this will write the
56       * "magic cookie" value.
57       *
58       *  Writing a version marker will reset the symbol table
59       *  to be the system symbol table.
60       */
61      void writeIonVersionMarker() throws IOException;
62  
63      /** Indicates whether the writer is stream copy optimized through {@link #writeValue(com.amazon.ion.IonReader)}. */
64      public boolean isStreamCopyOptimized();
65  }
