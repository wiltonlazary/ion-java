15  package software.amazon.ion.impl;

17  import static software.amazon.ion.impl.PrivateUtils.symtabExtends;
18  
19  import software.amazon.ion.SymbolTable;
20  
21  /**
22   * Cache to reduce unnecessary calls to
23   * {@link PrivateUtils#symtabExtends(SymbolTable, SymbolTable)}. This is
24   * only used if the writer is stream copy optimized.
25   * @deprecated This is an internal API that is subject to change without notice.
26   */
27  @Deprecated
28  public final class PrivateSymtabExtendsCache
29  {
30      private SymbolTable myWriterSymtab;
31      private SymbolTable myReaderSymtab;
32      private int myWriterSymtabMaxId;
33      private int myReaderSymtabMaxId;
34      private boolean myResult;
35  
36      public boolean symtabsCompat(SymbolTable writerSymtab,
37                                   SymbolTable readerSymtab)
38      {
39          // If the refs. of both writer's and reader's symtab match and are
40          // not modified, skip expensive symtab extends check and return
41          // cached result.
42  
43          assert writerSymtab != null && readerSymtab != null:
44              "writer's and reader's current symtab cannot be null";
45  
46          if (myWriterSymtab          == writerSymtab &&
47              myReaderSymtab          == readerSymtab &&
48              myWriterSymtabMaxId     == writerSymtab.getMaxId() &&
49              myReaderSymtabMaxId     == readerSymtab.getMaxId())
50          {
51              // Not modified, return cached result
52              return myResult;
53          }
54  
55          myResult = symtabExtends(writerSymtab, readerSymtab);
56  
57          // Track refs.
58          myWriterSymtab = writerSymtab;
59          myReaderSymtab = readerSymtab;
60  
61          // Track modification
62          myWriterSymtabMaxId = writerSymtab.getMaxId();
63          myReaderSymtabMaxId = readerSymtab.getMaxId();
64  
65          return myResult;
66      }
67  }
