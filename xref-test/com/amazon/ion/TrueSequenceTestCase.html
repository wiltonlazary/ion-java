95      public void testIndexedAddAll(IonSequence seq)
96      {
97          try
98          {
99              seq.addAll(0, null);
100             fail("expected exception");
101         }
102         catch (NullPointerException e) { }
103 
104         List<IonValue> list = new ArrayList<IonValue>();
105         try
106         {
107             seq.addAll(1, list);
108             fail("expected exception");
109         }
110         catch (IndexOutOfBoundsException e) { }
111 
112         boolean changed = seq.addAll(0, list);
113         assertFalse(changed);
114         assertEquals(0, seq.size());
115 
116         IonValue v1 = system().newInt(1);
117         list.add(v1);
118 
119         changed = seq.addAll(0, list);
120         assertTrue(changed);
121         assertEquals(1, seq.size());
122         assertSame(seq.get(0), v1);
123         assertSame(seq, v1.getContainer());
124 
125         try
126         {
127             seq.addAll(0, list);
128             fail("expected exception");
129         }
130         catch (ContainedValueException e) { }
131 
132         list.clear();
133         IonValue v2 = system().newInt(2);
134         IonValue v3 = system().newInt(3);
135         list.add(v2);
136         list.add(v3);
137 
138         seq.addAll(1, list);
139         assertEquals(3, seq.size());
140         assertSame(seq.get(0), v1);
141         assertSame(seq.get(1), v2);
142         assertSame(seq.get(2), v3);
143 
144         list.clear();
145         IonValue v4 = system().newInt(4);
146         list.add(v4);
147 
148         seq.addAll(1, list);
149         assertEquals(4, seq.size());
150         assertSame(seq.get(0), v1);
151         assertSame(seq.get(1), v4);
152         assertSame(seq.get(2), v2);
153         assertSame(seq.get(3), v3);
154     }

188     private void set(IonSequence s, int index)
189     {
190         IonValue origElement = s.get(index);
191         IonValue newElement = system().newInt(index);
192 
193         IonSequence expectedElements = s.clone();
194         expectedElements.remove(index);
195         expectedElements.add(index, newElement.clone());
196 
197         IonValue removed = s.set(index, newElement);
198 
199         assertSame(newElement, s.get(index));
200         assertSame(s, newElement.getContainer());
201         assertSame(origElement, removed);
202         assertEquals(null, removed.getContainer());
203 
204         assertEquals(index, ((_Private_IonValue)newElement).getElementId());
205 
206         assertEquals(expectedElements, s);
207     }

310     private <T extends RuntimeException>
311     void testSetThrows(IonSequence s, int index, IonValue v,
312                        Class<T> exceptionType)
313         throws RuntimeException
314     {
315         IonSequence orig = s.clone();
316         assertEquals(orig.isNullValue(), s.isNullValue());
317         try {
318             s.set(index, v);
319             fail("expected " + exceptionType.getSimpleName());
320         }
321         catch (RuntimeException e)
322         {
323             if (! exceptionType.isAssignableFrom(e.getClass()))
324             {
325                 throw e;
326             }
327         }
328         assertEquals(orig, s);
329     }
