19  @RunWith(JUnitParamsRunner.class)
20  public class IonSequenceLiteSubListViewEqualityTests {
21      static final IonSystem SYSTEM = IonSystemBuilder.standard().build();
22      private static List<Integer> INTS = Arrays.asList(0, 1, 2, 3, 4, 5, 6);
23      private static List<Integer> OTHER_INTS = Arrays.asList(0, 1, 2, 33, 4, 5, 6);
24      private static List<IonValue> ARRAY_LIST = newArrayList(INTS);
25      private static List<IonValue> OTHER_ARRAY_LIST = newArrayList(OTHER_INTS);
26  
27      private interface SublistMaker {
28          List<IonValue> makeSublist(List<IonValue> seq);
29      }
30  
31      private static void runSubListEquivalenceTests(List<IonValue> seq, SublistMaker maker) {
32          testSublistEquivalence(seq, maker);
33          testSublistNonEquivalence(seq, maker);
34      }
35  
36      // makes a sublist of seq and asserts its equivalence to sublists of all sequence types and to a sublist
37      // derived from ARRAY_LIST
38      private static void testSublistEquivalence(List<IonValue> seq, SublistMaker maker) {
39          List<IonValue> subList = maker.makeSublist(seq);
40          for(IonSequence otherSeq : getTestSequences()) {
41              List<IonValue> otherSubList = maker.makeSublist(otherSeq);
42              assertEquals("subLists should be equivalent", subList, otherSubList);
43              assertEquals("subLists should have the same hashCode", subList.hashCode(), otherSubList.hashCode());
44          }
45  
46          List<IonValue> ararySubList = maker.makeSublist(ARRAY_LIST);
47          assertEquals("subList should be equivalent to a sublist of ARRAY_LIST",
48                  ararySubList, subList);
49  
50          assertEquals("subList should have the same hash code as a sublist of ARRAY_LIST",
51                  ararySubList.hashCode(), subList.hashCode());
52      }
53  
54      // makes a sublist of seq and asserts its non-equivalence to the same subList range of the "other" sequence
55      // and to a sublist derived from OTHER_ARRAY_LIST
56      private static void testSublistNonEquivalence(List<IonValue> seq, SublistMaker maker) {
57          // A hash collision is unlikely but possible so we do not assert that the hashCodes are different here.
58  
59          List<IonValue> subList = maker.makeSublist(seq);
60          for(IonSequence otherSeq : getOtherTestSequences()) {
61              List<IonValue> otherSubList = maker.makeSublist(otherSeq);
62              assertNotEquals("subLists should *not* be equivalent", subList, otherSubList);
63          }
64  
65          assertNotEquals("subList should *not* be equivalent to a sublist of OTHER_ARRAY_LIST",
66                  maker.makeSublist(OTHER_ARRAY_LIST), subList);
67      }

69      private static IonSequence[] getParameters(List<Integer> ints) {
70          return new IonSequence[] {
71              populateSequence(SYSTEM.newList(), ints),
72              populateSequence(SYSTEM.newSexp(), ints),
73              populateSequence(SYSTEM.newDatagram(), ints)
74          };
75      }

78      @Test
79      @Parameters(method = "getTestSequences")
80      public void fullSubList(IonSequence seq) {
81          SublistMaker maker = new SublistMaker() {
82              @Override
83              public List<IonValue> makeSublist(List<IonValue> seq) {
84                  return seq.subList(0, seq.size());
85              }
86          };
87          runSubListEquivalenceTests(seq, maker);
88  
89          // The sublist should also be equivalent to the ARRAY_LIST
90          List<IonValue> subList = maker.makeSublist(seq);
91          assertEquals("The full sublist should be equivalent to the ArrayList<IonValue",
92                  ARRAY_LIST, subList);
93  
94          assertNotEquals("The full sublist should *not* be equivalent to the *other* ArrayList<IonValue",
95                  OTHER_ARRAY_LIST, subList);
96      }

101         SublistMaker maker = new SublistMaker() {
102             @Override

104                 return seq.subList(0, seq.size()).subList(0, seq.size());

134     @Test
135     @Parameters(method = "getTestSequences")
136     public void sequenceEquivalenceToSubList(IonSequence seq) {
137         List<IonValue> subList = seq.subList(0, seq.size());
138 
139         // Note the asymmetry.  This is expected. See IonSequence.subList javadoc.
140         assertEquals("the sublist should be equivalent to the sequence", subList, seq);
141         assertNotEquals("the sequence should *not* be equivalent to the subList", seq, subList);
142     }

166     private static IonSequence populateSequence(IonSequence seq, List<Integer> ints) {
167         for(int i : ints) {
168             seq.add(SYSTEM.newInt(i));
169         }
170         return seq;
171     }

173     private static ArrayList<IonValue> newArrayList(List<Integer> ints) {
174         ArrayList<IonValue> values = new ArrayList<IonValue>();
175         for(int i : ints) {
176             values.add(SYSTEM.newInt(i));
177         }
178         return values;
