16  package com.amazon.ion;
17  
18  import java.io.Reader;
19  import java.nio.charset.Charset;
20  
21  /**
22   * An Ion <code>clob</code> value.
23   * <p>
24   * <b>WARNING:</b> This interface should not be implemented or extended by
25   * code outside of this library.
26   */
27  public interface IonClob
28      extends IonValue, IonLob
29  {
30      /**
31       * Creates a new {@link Reader} that provides the value of this clob as
32       * text, decoding the raw bytes using a given character set.
33       *
34       * @param cs must not be <code>null</code>.
35       * @return a new reader positioned at the start of the clob,
36       * or <code>null</code> if <code>this.isNullValue()</code>.
37       */
38      public Reader newReader(Charset cs);
39  
40  
41      /**
42       * Gets the value of this clob as a Java {@link String} value, decoding
43       * the raw bytes using a given character set.  This is a convenience
44       * wrapper around {@link #newReader(Charset)}.
45       * <p>
46       * The behavior of this method when the clob bytes are not valid in the
47       * given charset is unspecified.
48       *
49       * @param cs must not be <code>null</code>.
50       * @return the decoded text,
51       * or <code>null</code> if <code>this.isNullValue()</code>.
52       */
53      public String stringValue(Charset cs);
54  
55  
56      public IonClob clone()
57          throws UnknownSymbolException;
58  }
