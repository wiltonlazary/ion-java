41      /**
42       * Parses text as a single Ion value.  If the text contains more than that,
43       * a failure is thrown.
44       *
45       * @param text must not be <code>null</code>.
46       * @return a single value, not <code>null</code>.
47       */
48      public IonValue loadOneValue(String text)
49      {
50          IonDatagram dg = loader().load(text);
51  
52          if (dg.size() == 0)
53          {
54              fail("No user values in text: " + text);
55          }
56  
57          if (dg.size() > 1)
58          {
59              IonValue part = dg.get(1);
60              fail("Found unexpected part <" + part + "> in text: " + text);
61          }
62  
63          IonValue value = dg.get(0);
64          dg.remove(value);
65          return value;
66      }

396     @Test
397     public void testCatalogOnLoader() throws Exception {
398         IonSystem sys = newSystem(Symtabs.CATALOG);
399         IonDatagram dg = sys.newDatagram(Symtabs.CATALOG.getTable("fred", 1));
400         dg.add().newSymbol("fred_1");
401         byte[] raw = dg.getBytes();
402         IonLoader loader = sys.newLoader();
403         IonDatagram dg_fred = loader.load(raw);
404         IonValue v = dg_fred.get(0);
405         IonSymbol sym_fred = (IonSymbol) v;
406         String string_fred = sym_fred.stringValue();
407         assertEquals("fred_1", string_fred);
408 
409         // let's make an interesting catalog that has
410         // a different v1 fred symtab
411         IonMutableCatalog newCatalog = new SimpleCatalog();
412         newCatalog.putTable(sys.newSharedSymbolTable("fred", 1, Arrays.asList("barney_1").iterator()));
413         IonLoader new_loader = sys.newLoader(newCatalog);
414         IonDatagram new_dg = new_loader.load(raw);
415         IonSymbol sym = (IonSymbol)new_dg.get(0);
416         checkSymbol("barney_1", sym);
417     }

421     @Test
422     public void testIteratingVeryLongFile()
423     throws Exception
424     {
425     	TestTimer gc = new TestTimer("gc()", 10, 5);
426     	TestTimer read = new TestTimer("read", 10, 2);
427     	long loop_counter = 0;
428     	IonValue value = null;
429     	boolean  more;
430 
431 // TODO: turn this long test on someday
432 if (!_debug_long_test) return;
433 
434 	LongInputStream longStream = new LongInputStream("$ion_1_0 { this:is, a:struct } ");
435         Reader reader = new InputStreamReader(longStream, UTF8_CHARSET);
436         Iterator<IonValue> i = system().iterate(reader);
437 
438         for (;;)
439         {
440         	read.start();
441         	more = i.hasNext();
442             if (more) {
443             	value = i.next();
444             }
445             read.finish(loop_counter);
446 
447             assertFalse(value == null || value.isNullValue());
448             assertNull(value.getContainer());
449             //assertFalse(i.isNullValue());
450             //assertFalse(i.isInStruct());
451 
452             loop_counter++;
453             if (loop_counter % 1000 == 0) {
454             	System.out.println("counter = "+loop_counter+", moving average = "+read.getAverage());
455             }
456             if (loop_counter > 100000) break;
457 
458         	gc.start();
459         	System.gc();
460         	gc.finish(loop_counter);
461         }
462     }

464     static class TestTimer {
465     	String _title;
466     	long[] _samples;
467     	int    _next;
468     	int    _sample_length;
469     	long   _moving_average;  // really the moving total
470     	long   _threshold;
471     	int    _multiplier;
472     	long   _start;
473 
474     	TestTimer(String title, int length, int multiplier) {
475     		this._title = title;
476     		this._samples = new long[length];
477     		this._sample_length = length;
478     		this._moving_average = 0;
479     		this._threshold = -1;
480     		this._multiplier = multiplier;
481     	}
482     	void reset() {
483     		this._next = 0;
484     		this._threshold = -1;
485     		this._moving_average = 0;
486     		for (int ii=0; ii<this._sample_length; ii++) {
487     			this._samples[ii] = 0;
488     		}
489     	}
490     	long getAverage() {
491     		if (this._threshold < 0) return -1;
492     		return this._moving_average / this._sample_length;
493     	}
494     	void start() {
495     		this._start = System.nanoTime();
496     	}
497     	void finish(long counter) {
498     		long end = System.nanoTime();
499     		long duration = end - this._start;
500 
501     		if (this._next == this._sample_length) {
502     			if (this._threshold == -1) {
503     				long temp = 0;
504     				for (int ii=0; ii<this._sample_length; ii++) {
505     					temp += this._samples[ii];
506     				}
507     				this._moving_average = temp;
508     				this._threshold = temp / this._sample_length;
509     				this._threshold *= this._multiplier;
510     			}
511     			this._next = 0;
512     		}
513     		this._moving_average -= this._samples[this._next];
514     		this._moving_average += duration;
515     		this._samples[this._next] =  duration;
516     		this._next++;
517     		if ((this._moving_average/this._sample_length) > this._threshold) {
518     			if (this._threshold != -1) {
519     				System.out.println("it's taking too long to "+this._title+" at counter = "+counter+". moving average = "+this._moving_average+", timer threshold = "+this._threshold);
520     				this.reset();
521     			}
522     		}
523     	}
524     }

526     static class LongInputStream extends InputStream
527     {
528     	byte[] _pattern;
529     	int    _pos;
530     	int    _limit;
531 
532     	LongInputStream(String pattern) {
533     	    this._pattern = utf8(pattern);
534     	    this._limit = _pattern.length;
535     	    this._pos = 0;
536     	}
537 
538     	@Override
539     	public int read()
540     	{
541     	    if (this._pos >= this._limit) {
542     	        this._pos = 0;
543     	    }
544     	    int c = this._pattern[this._pos];
545     	    this._pos++;
546     	    return c;
547     	}
