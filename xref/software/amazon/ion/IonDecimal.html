15  package software.amazon.ion;
16  
17  import java.math.BigDecimal;
18  
19  /**
20   * An Ion <code>decimal</code> value.
21   * <p>
22   * <b>WARNING:</b> This interface should not be implemented or extended by
23   * code outside of this library.
24   */
25  public interface IonDecimal
26      extends IonNumber
27  {
28      /**
29       * Gets the value of this Ion {@code decimal} as a Java
30       * <code>float</code> value.
31       *
32       * @return the float value.
33       * @throws NullValueException if <code>this.isNullValue()</code>.
34       */
35      public float floatValue()
36          throws NullValueException;
37  
38      /**
39       * Gets the value of this Ion {@code decimal} as a Java
40       * <code>double</code> value.
41       *
42       * @return the double value.
43       * @throws NullValueException if <code>this.isNullValue()</code>.
44       */
45      public double doubleValue()
46          throws NullValueException;
47  
48      /**
49       * Gets the value of this Ion {@code decimal} as a {@link BigDecimal}.
50       * If you need negative zeros, use {@link #decimalValue()}.
51       *
52       * @return the {@code BigDecimal} value,
53       * or {@code null} if {@code this.isNullValue()}.
54       *
55       * @see #decimalValue()
56       */
57      public BigDecimal bigDecimalValue();
58  
59      /**
60       * Gets the value of this Ion {@code decimal} as a {@link Decimal},
61       * which extends {@link BigDecimal} with support for negative zeros.
62       *
63       * @return the {@code Decimal} value,
64       * or {@code null} if {@code this.isNullValue()}.
65       *
66       * @see #bigDecimalValue()
67       */
68      public Decimal decimalValue();
69  
70      /**
71       * Sets the value of this element.
72       */
73      public void setValue(long value);
74  
75      /**
76       * Sets the value of this element.
77       * This method behaves like {@link BigDecimal#valueOf(double)} in that it
78       * uses the {@code double}'s canonical string representation provided by
79       * {@link Double#toString(double)}}.
80       *
81       * @see Decimal#valueOf(double)
82       */
83      public void setValue(float value);
84  
85      /**
86       * Sets the value of this element.
87       * This method behaves like {@link BigDecimal#valueOf(double)} in that it
88       * uses the {@code double}'s canonical string representation provided by
89       * {@link Double#toString(double)}}.
90       *
91       * @see Decimal#valueOf(double)
92       */
93      public void setValue(double value);
94  
95      /**
96       * Sets the value of this element.
97       * To set a negative zero value, pass an {@link Decimal}.
98       *
99       * @param value the new value of this decimal;
100      * may be <code>null</code> to make this <code>null.decimal</code>.
101      */
102     public void setValue(BigDecimal value);
103 
104 
105     public IonDecimal clone()
106         throws UnknownSymbolException;
107 }
