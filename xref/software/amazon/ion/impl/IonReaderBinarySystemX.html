92      protected final void prepare_value(int as_type) {
93          load_once();
94          if (as_type != 0 && !_v.hasValueOfType(as_type)) {
95              // we should never get here with a symbol asking for anything other
96              // than a numeric cast (from some other numeric already loaded)
97              if (IonType.SYMBOL.equals(_value_type) && !ValueVariant.isNumericType(as_type)) {
98                  assert(IonType.SYMBOL.equals(_value_type) && !ValueVariant.isNumericType(as_type));
99              }
100 
101             if (!_v.can_convert(as_type)) {
102                 String message = "can't cast from "
103                     +PrivateScalarConversions.getValueTypeName(_v.getAuthoritativeType())
104                     +" to "
105                     +PrivateScalarConversions.getValueTypeName(as_type);
106                 throw new IllegalStateException(message);
107             }
108             int fnid = _v.get_conversion_fnid(as_type);
109             _v.cast(fnid);
110         }
111     }

131     /** Utility method to convert an unsigned magnitude stored as a long to a {@link BigInteger}. */
132     private static BigInteger unsignedLongToBigInteger(int signum, long val)
133     {
134         byte[] magnitude = {
135             (byte) ((val >> 56) & 0xFF),
136             (byte) ((val >> 48) & 0xFF),
137             (byte) ((val >> 40) & 0xFF),
138             (byte) ((val >> 32) & 0xFF),
139             (byte) ((val >> 24) & 0xFF),
140             (byte) ((val >> 16) & 0xFF),
141             (byte) ((val >>  8) & 0xFF),
142             (byte) (val & 0xFF),
143         };
144         return new BigInteger(signum, magnitude);
145     }

147     static final BigInteger MIN_LONG_VALUE = BigInteger.valueOf(Long.MIN_VALUE);
148     static final BigInteger MAX_LONG_VALUE = BigInteger.valueOf(Long.MAX_VALUE);
149 
150     private final void load_scalar_value() throws IOException
151     {
152         // make sure we're trying to load a scalar value here
153         switch(_value_type) {
154         case NULL:
155         case BOOL:
156         case INT:
157         case FLOAT:
158         case DECIMAL:
159         case TIMESTAMP:
160         case SYMBOL:
161         case STRING:
162             break;
163         default:
164             return;
165         }
166 
167         // this will be true when the value_type is null as
168         // well as when we encounter a null of any other type
169         if (_value_is_null) {
170             _v.setValueToNull(_value_type);
171             _v.setAuthoritativeType(AS_TYPE.null_value);
172             return;
173         }
174 
175         switch (_value_type) {
176         default:
177             return;
178         case BOOL:
179             _v.setValue(_value_is_true);
180             _v.setAuthoritativeType(AS_TYPE.boolean_value);
181             break;
182         case INT:
183             boolean is_negative = _value_tid == PrivateIonConstants.tidNegInt;
184 
185             if (_value_len == 0) {
186                 if (is_negative) {
187                     throwIllegalNegativeZeroException();
188                 }
189 
190                 int v = 0;
191                 _v.setValue(v);
192                 _v.setAuthoritativeType(AS_TYPE.int_value);
193             }
194             else if (_value_len <= Long.BYTES) {
195                 long v = readULong(_value_len);
196 
197                 if (v < 0) {
198                     // we probably can't fit this magnitude properly into a Java long
199                     int signum = !is_negative ? 1 : -1;
200                     BigInteger big = unsignedLongToBigInteger(signum, v);
201                     _v.setValue(big);
202                     // boundary condition
203                     if (big.compareTo(MIN_LONG_VALUE) < 0 || big.compareTo(MAX_LONG_VALUE) > 0) {
204                         _v.setAuthoritativeType(AS_TYPE.bigInteger_value);
205                     } else {
206                         // fits in long
207                         _v.addValue(big.longValue()); // keep the BigInteger value set in case the user wants to resurrect it as such
208                         _v.setAuthoritativeType(AS_TYPE.long_value);
209                     }
210                 }
211                 else {
212                     if (is_negative) {
213                         if(v == 0) {
214                             throwIllegalNegativeZeroException();
215                         }
216                         v = -v;
217                     }
218                     if (v < Integer.MIN_VALUE || v > Integer.MAX_VALUE) {
219                         _v.setValue(v);
220                         _v.setAuthoritativeType(AS_TYPE.long_value);
221                     } else {
222                         _v.setValue((int)v);
223                         _v.setAuthoritativeType(AS_TYPE.int_value);
224                     }
225                 }
226             }
227             else {
228                 BigInteger v = readBigInteger(_value_len, is_negative);
229                 _v.setValue(v);
230                 _v.setAuthoritativeType(AS_TYPE.bigInteger_value);
231             }
232             break;
233         case FLOAT:
234             double d;
235             if (_value_len == 0) {
236                 d = 0.0;
237             }
238             else {
239                 d = readFloat(_value_len);
240             }
241             _v.setValue(d);
242             _v.setAuthoritativeType(AS_TYPE.double_value);
243             break;
244         case DECIMAL:
245             Decimal dec = readDecimal(_value_len);
246             _v.setValue(dec);
247             _v.setAuthoritativeType(AS_TYPE.decimal_value);
248             break;
249         case TIMESTAMP:
250             // TODO: it looks like a 0 length return a null timestamp - is that right?
251             Timestamp t = readTimestamp(_value_len);
252             _v.setValue(t);
253             _v.setAuthoritativeType(AS_TYPE.timestamp_value);
254             break;
255         case SYMBOL:
256             long sid = readULong(_value_len);
257             if (sid < 0 || sid > Integer.MAX_VALUE) {
258                 String message = "symbol id ["
259                                + sid
260                                + "] out of range "
261                                + "(1-"
262                                + Integer.MAX_VALUE
263                                + ")";
264                 throwErrorAt(message);
265             }
266             // TODO: is treating this as an int too misleading?
267             _v.setValue((int)sid);
268             _v.setAuthoritativeType(AS_TYPE.int_value);
269             break;
270         case STRING:
271             String s = readString(_value_len);
272             _v.setValue(s);
273             _v.setAuthoritativeType(AS_TYPE.string_value);
274             break;
275         }
276         _state = State.S_AFTER_VALUE;
277     }

392     public final String stringValue()
393     {
394         if (! IonType.isText(_value_type)) throw new IllegalStateException();
395         if (_value_is_null) return null;
396 
397         if (_value_type == SYMBOL) {
398             if (!_v.hasValueOfType(AS_TYPE.string_value)) {
399                 int sid = getSymbolId();
400                 String name = _symbols.findKnownSymbol(sid);
401                 if (name == null) {
402                     throw new UnknownSymbolException(sid);
403                 }
404                 _v.addValue(name);
405             }
406         }
407         else {
408             prepare_value(AS_TYPE.string_value);
409         }
410         return _v.getString();
411     }

413     public SymbolToken symbolValue()
414     {
415         if (_value_type != SYMBOL) throw new IllegalStateException();
416         if (_value_is_null) return null;

420         String text = _symbols.findKnownSymbol(sid);

434     public final String getFieldName()
435     {
436         String name;
437         if (_value_field_id == SymbolTable.UNKNOWN_SYMBOL_ID) {
438             name = null;
439         }
440         else {
441             name = _symbols.findKnownSymbol(_value_field_id);
442             if (name == null) {
443                 throw new UnknownSymbolException(_value_field_id);
444             }
445         }
446         return name;

463     public final String[] getTypeAnnotations()
464     {
465         load_annotations();
466         String[] anns;
467         if (_annotation_count < 1) {
468             anns = PrivateUtils.EMPTY_STRING_ARRAY;
469         }
470         else {
471             anns = new String[_annotation_count];
472             for (int ii=0; ii<_annotation_count; ii++) {
473                 anns[ii] = _symbols.findKnownSymbol(_annotation_ids[ii]);
474                 if (anns[ii] == null) {
475                     throw new UnknownSymbolException(_annotation_ids[ii]);
476                 }
477             }
478         }
479         return anns;
480     }
