209     /**
210      * Parses the symbol table at the reader's current position.
211      * @param reader the reader from which to parse the symbol table.
212      * @param catalog the catalog from which to resolve shared symbol table imports.
213      * @param isOnStruct true if the reader is already positioned on the symbol table struct; otherwise, false.
214      * @param symbolsListOut list into which local symbols declared by the parsed symbol table will be deposited.
215      * @param currentSymbolTable the symbol table currently active in the stream.
216      * @return a new LocalSymbolTableImports instance, or null if this was an LST append. If null, `currentSymbolTable`
217      *   continues to be the active symbol table.
218      */
219     protected static LocalSymbolTableImports readLocalSymbolTable(IonReader reader,
220                                                                   IonCatalog catalog,
221                                                                   boolean isOnStruct,
222                                                                   List<String> symbolsListOut,
223                                                                   SymbolTable currentSymbolTable)
224     {
225         if (! isOnStruct)
226         {
227             reader.next();
228         }
229 
230         assert reader.getType() == IonType.STRUCT
231             : "invalid symbol table image passed in reader " +
232               reader.getType() + " encountered when a struct was expected";
233 
234         assert ION_SYMBOL_TABLE.equals(reader.getTypeAnnotations()[0])
235             : "local symbol tables must be annotated by " + ION_SYMBOL_TABLE;
236 
237         reader.stepIn();
238 
239         List<SymbolTable> importsList = new ArrayList<SymbolTable>();
240         importsList.add(reader.getSymbolTable().getSystemSymbolTable());
241 
242         IonType fieldType;
243         boolean foundImportList = false;
244         boolean foundLocalSymbolList = false;
245         boolean isAppend = false;
246         while ((fieldType = reader.next()) != null)
247         {
248             if (reader.isNullValue()) continue;
249 
250             SymbolToken symTok = reader.getFieldNameSymbol();
251             int sid = symTok.getSid();
252             if (sid == SymbolTable.UNKNOWN_SYMBOL_ID)
253             {
254                 // This is a user-defined IonReader or a pure DOM, fall
255                 // back to text
256                 final String fieldName = reader.getFieldName();
257                 sid = getSidForSymbolTableField(fieldName);
258             }
259 
260             // TODO amzn/ion-java/issues/36 Switching over SIDs doesn't cover the case
261             //      where the relevant field names are defined by a prev LST;
262             //      the prev LST could have 'symbols' defined locally with a
263             //      different SID!
264             switch (sid)
265             {
266                 case SYMBOLS_SID:
267                 {
268                     // As per the Spec, other field types are treated as
269                     // empty lists
270                     if(foundLocalSymbolList){
271                         throw new IonException("Multiple symbol fields found within a single local symbol table.");
272                     }
273                     foundLocalSymbolList = true;
274                     if (fieldType == IonType.LIST)
275                     {
276                         reader.stepIn();
277                         IonType type;
278                         while ((type = reader.next()) != null)
279                         {
280                             final String text;
281                             if (type == IonType.STRING)
282                             {
283                                 text = reader.stringValue();
284                             }
285                             else
286                             {
287                                 text = null;
288                             }
289 
290                             symbolsListOut.add(text);
291                         }
292                         reader.stepOut();
293                     }
294                     break;
295                 }
296                 case IMPORTS_SID:
297                 {
298                     if(foundImportList){
299                         throw new IonException("Multiple imports fields found within a single local symbol table.");
300                     }
301                     foundImportList = true;
302                     if (fieldType == IonType.LIST)
303                     {
304                         prepImportsList(importsList, reader, catalog);
305                     }
306                     else if (fieldType == IonType.SYMBOL && ION_SYMBOL_TABLE.equals(reader.stringValue()))
307                     {
308                         isAppend = true;
309                     }
310                     break;
311                 }
312                 default:
313                 {
314                     // As per the Spec, any other field is ignored
315                     break;
316                 }
317             }
318         }
319         reader.stepOut();
320         if (isAppend && currentSymbolTable.isLocalTable()) {
321             // Because the current symbol table is a local symbol table (i.e. not the system symbol table), it can
322             // be appended in-place.
323             LocalSymbolTableimpl/LocalSymbolTable.html#LocalSymbolTable">LocalSymbolTable currentLocalSymbolTable = (LocalSymbolTable) currentSymbolTable;
324             for (String newSymbol : symbolsListOut) {
325                 currentLocalSymbolTable.putSymbol(newSymbol);
326             }
327             return null;
328         }
329         return new LocalSymbolTableImports(importsList);
330     }

479     public SymbolToken find(String text)
480     {
481         text.getClass(); // fast null check
482 
483         // Look in system then imports
484         SymbolToken symTok = myImportsList.find(text);
485 
486         // Look in local symbols
487         if (symTok == null)
488         {
489             Integer  sid;
490             String[] names;
491             synchronized (this)

503             }
504         }

509     private static final void validateSymbol(String name)
510     {
511         if (name == null)
512         {
513             throw new IllegalArgumentException("symbols must not be null");
514         }
515         for (int i = 0; i < name.length(); ++i)
516         {
517             int c = name.charAt(i);
518             if (c >= 0xD800 && c <= 0xDFFF)
519             {
520                 if (c >= 0xDC00)
521                 {
522                     String message = "unpaired trailing surrogate in symbol " +
523                             "name at position " + i;
524                     throw new IllegalArgumentException(message);
525                 }
526                 ++i;
527                 if (i == name.length())
528                 {
529                     String message = "unmatched leading surrogate in symbol " +
530                             "name at position " + i;
531                     throw new IllegalArgumentException(message);
532                 }
533                 c = name.charAt(i);
534                 if (c < 0xDC00 || c > 0xDFFF)
535                 {
536                     String message = "unmatched leading surrogate in symbol " +
537                             "name at position " + i;
538                     throw new IllegalArgumentException(message);
539                 }
540             }
541         }
542     }

544     /**
545      * NOT SYNCHRONIZED! Call within constructor or from synch'd method.
546      */
547     int putSymbol(String symbolName)
548     {
549 
550         if (isReadOnly)
551         {
552             throw new ReadOnlyValueException(SymbolTable.class);
553         }
554 
555         if (mySymbolsCount == mySymbolNames.length)
556         {
557             int newlen = mySymbolsCount * 2;
558             if (newlen < DEFAULT_CAPACITY)
559             {
560                 newlen = DEFAULT_CAPACITY;
561             }
562             String[] temp = new String[newlen];
563             System.arraycopy(mySymbolNames, 0, temp, 0, mySymbolsCount);
564             mySymbolNames = temp;
565         }
566 
567         int sid = -1;
568         if (symbolName != null)
569         {
570             sid = mySymbolsCount + myFirstLocalSid;
571             assert sid == getMaxId() + 1;
572 
573             putToMapIfNotThere(mySymbolsMap, symbolName, sid);
574         }
575         mySymbolNames[mySymbolsCount] = symbolName;
576         mySymbolsCount++;
577 
578         return sid;
579     }

581     private static void putToMapIfNotThere(Map<String, Integer> symbolsMap,
582                                            String text,
583                                            int sid)
584     {
585         // When there's a duplicate name, don't replace the lower sid.
586         // This pattern avoids double-lookup in the normal happy case
587         // and only requires a second lookup when there's a duplicate.
588         Integer extantSid = symbolsMap.put(text, sid);
589         if (extantSid != null)
590         {
591             // We always insert symbols with increasing sids
592             assert extantSid < sid;
593             symbolsMap.put(text, extantSid);
594         }
595     }

657     /**
658      * Returns a {@link SymbolTable} representation of a single import
659      * declaration from the passed-in reader and catalog.
660      *
661      * @return
662      *          symbol table representation of the import; null if the import
663      *          declaration is malformed
664      */
665     private static SymbolTable readOneImport(IonReader ionRep,
666                                              IonCatalog catalog)
667     {
668         assert (ionRep.getType() == IonType.STRUCT);
669 
670         String name = null;
671         int    version = -1;
672         int    maxid = -1;
673 
674         ionRep.stepIn();
675         IonType t;
676         while ((t = ionRep.next()) != null)
677         {
678             if (ionRep.isNullValue()) continue;
679 
680             SymbolToken symTok = ionRep.getFieldNameSymbol();
681             int field_id = symTok.getSid();
682             if (field_id == UNKNOWN_SYMBOL_ID)
683             {
684                 // this is a user defined reader or a pure DOM
685                 // we fall back to text here
686                 final String fieldName = ionRep.getFieldName();
687                 field_id = getSidForSymbolTableField(fieldName);
688             }
689             switch(field_id)
690             {
691                 case NAME_SID:
692                     if (t == IonType.STRING)
693                     {
694                         name = ionRep.stringValue();
695                     }
696                     break;
697                 case VERSION_SID:
698                     if (t == IonType.INT)
699                     {
700                         version = ionRep.intValue();
701                     }
702                     break;
703                 case MAX_ID_SID:
704                     if (t == IonType.INT)
705                     {
706                         maxid = ionRep.intValue();
707                     }
708                     break;
709                 default:
710                     // we just ignore anything else as "open content"
711                     break;
712             }
713         }
714         ionRep.stepOut();
715 
716         // Ignore import clauses with malformed name field.
717         if (name == null || name.length() == 0 || name.equals(ION))
718         {
719             return null;
720         }
721 
722         if (version < 1)
723         {
724             version = 1;
725         }
726 
727         SymbolTable itab = null;
728         if (catalog != null)
729         {
730             itab = catalog.getTable(name, version);
731         }
732         if (maxid < 0)
733         {
734             if (itab == null || version != itab.getVersion())
735             {
736                 String message =
737                     "Import of shared table "
738                         + IonTextUtils.printString(name)
739                         + " lacks a valid max_id field, but an exact match was not"
740                         + " found in the catalog";
741                 if (itab != null)
742                 {
743                     message += " (found version " + itab.getVersion() + ")";
744                 }
745                 // TODO custom exception
746                 throw new IonException(message);
747             }
748 
749             // Exact match is found, but max_id is undefined in import
750             // declaration, set max_id to largest sid of shared symtab
751             maxid = itab.getMaxId();
752         }
753 
754         if (itab == null)
755         {
756             assert maxid >= 0;
757 
758             // Construct substitute table with max_id undefined symbols
759             itab = new SubstituteSymbolTable(name, version, maxid);
760         }
761         else if (itab.getVersion() != version || itab.getMaxId() != maxid)
762         {
763             // A match was found BUT specs are not an exact match
764             // Construct a substitute with correct specs, containing the
765             // original import table that was found
766             itab = new SubstituteSymbolTable(itab, version, maxid);
767         }
768 
769         return itab;
770     }

854         if (mySymbolsCount < subLocalSymbolCount) return false;
