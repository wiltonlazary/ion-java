16  package com.amazon.ion;
17  
18  
19  /**
20   * An error caused by use of an invalid symbol starting with
21   * <code>"$ion_"</code>.
22   */
23  public class InvalidSystemSymbolException
24      extends IonException
25  {
26      private static final long serialVersionUID = 2206499395645594047L;
27  
28      private String myBadSymbol;
29  
30  
31      public InvalidSystemSymbolException(String badSymbol)
32      {
33          super("Invalid system symbol '" + badSymbol + "'");
34          myBadSymbol = badSymbol;
35      }
36  
37  
38      public String getBadSymbol()
39      {
40          return myBadSymbol;
41      }
42  }
