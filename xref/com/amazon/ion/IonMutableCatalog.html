18  /**
19   * An {@link IonCatalog} that can be updated.

21  public interface IonMutableCatalog extends IonCatalog {

23       * Adds a symbol table to this catalog.  This interface does not define the
24       * behavior of this method if this catalog already contains a table with
25       * the same name and version.

27       * @param sharedTable must be shared but not a system table or
28       *        substitute table.
29       */
30      public void putTable(SymbolTable sharedTable);
31  }
