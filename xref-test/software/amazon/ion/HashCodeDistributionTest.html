42  /** Chi-square test for {@link IonValue#hashCode()} implementations.
43   * Using selected test data, determine Chi^2 value for various hash table
44   * sizes and determine whether it is within 95% confidence interval using a
45   * goodness of fit test. The Chi-square test tests for the fact that hashes
46   * generated from the hash functions are random, with a 95% confidence.
47   * <p>
48   * See Knuth vol. 2 sec 3.3.1 for background.
49   */
50  @Ignore
51  public class HashCodeDistributionTest
52      extends IonTestCase
53  {
54      /**
55       * This value is the initial capacity of the data set (i.e. number of
56       * buckets). This must be a power of 2.
57       * <p>
58       * We want to keep halving the number of buckets in the data set to test on
59       * different sizes of buckets. The halving of the buckets is performed at
60       * {@link #collapse(int[])}.
61       * <p>
62       * The rationale behind this is that we want to mimic the internal
63       * implementation of how {@link HashMap} and {@link ConcurrentHashMap}
64       * stores its hash table, which is an {@link Entry} array with size that is
65       * a power of 2.
66       */
67      private static final int DATA_SET_INITIAL_CAPACITY = 1 << 12;
68  
69      /**
70       * This is sync'ed with {@link HashMap}'s DEFAULT_INITIAL_CAPACITY
71       * constant, which is 16.
72       */
73      private static final int HASH_MAP_INITIAL_CAPACITY = 1 << 4;
74  
75      private static final String BULK_TEST_DATA_PATH = "items-A";
76  
77      protected Set<IonValue> loadBulkDataSet()
78          throws IOException
79      {
80          Set<IonValue> dataSet = new HashSet<IonValue>();
81          File directory = getBulkTestdataFile(BULK_TEST_DATA_PATH);
82  
83          File[] testFiles = directory.listFiles(new FilenameFilter()
84          {
85              public boolean accept(File dir, String name)
86              {
87                  return name.endsWith(".ion") || name.endsWith(".10n");
88              }
89          });
90  
91          IonSystem ionSystem = system();
92          for (File testFile : testFiles)
93          {
94              InputStream in =
95                  new BufferedInputStream(new FileInputStream(testFile));
96              try
97              {
98                  Iterator<IonValue> i = ionSystem.iterate(in);
99                  while (i.hasNext())
100                 {
101                     loadValue(i.next(), dataSet);
102                 }
103             }
104             finally
105             {
106                 in.close();
107             }
108         }
109 
110         return dataSet;
111     }

113     /**
114      * Loads the {@link IonValue} and its nested values into the {@code dataSet}.
115      *
116      * @param value
117      * @param dataSet
118      */
119     protected void loadValue(IonValue value, Set<IonValue> dataSet)
120     {
121         dataSet.add(value);
122         if (value instanceof IonContainer)
123         {
124             IonContainer container = (IonContainer) value;
125             for (IonValue v : container)
126             {
127                 loadValue(v, dataSet);
128             }
129         }
130     }
131 
132     /**
133      * Analyze key entries, {@code counts}, against the Chi^2 distribution as
134      * provided, while collapsing {@code counts} down to {@code limit} from the
135      * initial size.
136      *
137      * @param counts key entries int array
138      * @param totalCount total number of key entries
139      * @param limit limit for {@link #collapse(int[])}
140      */
141     protected void analyzeDistribution(int[] counts, int totalCount, int limit)
142     {
143         while (counts.length >= limit)
144         {
145             checkChiSquareValue(counts, totalCount);
146             counts = collapse(counts);
147         }
148     }
149 
150     /**
151      * Analyze counts against Chi^2 distribution at 95% confidence interval.
152      *
153      * @param counts key entries int array
154      */
155     protected void checkChiSquareValue(int[] counts, int totalCount)
156     {
157         int expectedTotalCount = 0;
158         for (int count : counts)
159         {
160             expectedTotalCount += count;
161         }
162 
163         assertEquals("Total counts is incorrect", totalCount, expectedTotalCount);
164 
165         if (expectedTotalCount == 0)
166         {
167             return; // nothing to test
168         }
169 
170         double expectedValue = (double) totalCount / (double) counts.length;
171         double chiSquared = 0.0;
172         for (int count : counts)
173         {
174             chiSquared += Math.pow((count - expectedValue), 2) / expectedValue;
175         }
176         int degreesOfFreedom = counts.length - 1;
177 
178         double threshold;
179         if (degreesOfFreedom < upperTailChiSquared95Table.length)
180         {
181             threshold = upperTailChiSquared95Table[degreesOfFreedom];
182         }
183         else
184         {
185             // Calculate critical chi-square value using formula for degrees of
186             // freedom greater than 100. Refer to Knuth vol. 2 sec 3.3.1
187             double x_p = 1.64;
188             threshold = degreesOfFreedom
189                         + Math.sqrt(2*degreesOfFreedom)*x_p
190                         + (2.0/3.0)*Math.pow(x_p, 2)
191                         - (2.0/3.0);
192         }
193 
194         if (chiSquared >= threshold)
195         {
196             fail(String.format("Chi-square value above 95 percent " +
197                                "threshold: %1$f, limit: %2$f, nu: %3$d",
198                                chiSquared, threshold, degreesOfFreedom));
199         }
200     }
201 
202     /**
203      * Collapse key entries int array and return a new int array.
204      *
205      * @param counts
206      * @return the new key entries int array
207      */
208     protected int[] collapse(int[] counts)
209     {
210         int[] new_counts = new int[counts.length >> 1]; // halve the number of buckets
211         for (int i = 0; i < new_counts.length; ++i)
212         {
213             new_counts[i] = counts[i] + counts[i + new_counts.length];
214         }
215         return new_counts;
216     }

218     /**
219      * Duplicate protected method {@link HashMap#hash(int)}.  Formula from
220      * decompiled class in JDK 1.6.0.
221      * @param raw_hash Raw hash value
222      * @return cooked hash value
223      */
224     protected int hashMapHash(int raw_hash)
225     {
226         int cooked_hash = raw_hash;
227         cooked_hash ^= (cooked_hash >>> 20) ^ (cooked_hash >>> 12);
228         return cooked_hash ^ (cooked_hash >>> 7) ^ (cooked_hash >>> 4);
229     }

231     /**
232      * Counts each occurrence of {@link IonValue#hashCode()} and perform
233      * Chi^2 analysis over it.
234      *
235      * @param dataSet
236      *          the data set containing IonValue test data
237      * @param type
238      *          filter over IonValue test data; all IonValues are used if
239      *          {@code null}
240      * @param adjusted
241      *          {@code true} if using {@link #hashMapHash(int)} as a
242      *          supplementary hash function
243      * @param limit
244      *          the limit to {@link #collapse(int[])}
245      */
246     protected void checkChiSquareIonValueHashCode(Set<IonValue> dataSet,
247                                                   IonType type,
248                                                   boolean adjusted,
249                                                   int limit)
250     {
251         int[] counts = new int[DATA_SET_INITIAL_CAPACITY]; // must be power of 2
252         int countsLength = counts.length;
253         int totalCount = 0;
254         for (IonValue v : dataSet)
255         {
256             if (type == null || v.getType().equals(type))
257             {
258                 int index = adjusted ?
259                             hashMapHash(v.hashCode()) % countsLength :
260                             v.hashCode() % countsLength;
261                 if (index < 0)
262                 {
263                     index += countsLength;
264                 }
265                 counts[index]++;
266                 totalCount++;
267             }
268         }
269         analyzeDistribution(counts, totalCount, limit);
270     }
