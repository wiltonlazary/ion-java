16  package com.amazon.ion.impl.bin;
17  
18  import java.io.Closeable;
19  
20  
21  /**
22   * An abstraction for a block of managed memory.  A {@link Block} is acquired by a caller via
23   * {@link BlockAllocator#allocateBlock()} and released by {@link #close()}.
24   * <p>
25   * This class and its implementations are <b>not</b> thread-safe.
26   */
27  /*package*/ abstract class Block implements Closeable
28  {
29      /** The data backing this block. */
30      public final byte[] data;
31      /** The first index for which data has not been written to or read from. */
32      public int limit;
33  
34      /*package*/ Block(final byte[] data)
35      {
36          this.data = data;
37          this.limit = 0;
38      }
39  
40      /** Resets the limit to zero. */
41      public final void reset()
42      {
43          limit = 0;
44      }
45  
46      /** Returns the unused amount of bytes from the limit to the capacity of the data array. */
47      public final int remaining()
48      {
49          return data.length - limit;
50      }
51  
52      /** Returns the underlying data array's capacity. */
53      public final int capacity()
54      {
55          return data.length;
56      }
57  
58      /**
59       * Releases the block back to the {@link BlockAllocator} from whence it was allocated from.
60       * This method <b>must</b> be called when the block is no longer needed.
61       * This is used to allow underlying resources to be reclaimed and/or released properly.
62       */
63      public abstract void close();
64  }
