16  package com.amazon.ion;
17  
18  public class BitUtils
19  {
20      public static byte[] bytes(int... ints)
21      {
22          byte[] b = new byte[ints.length];
23          for (int i = 0; i < ints.length; i++)
24          {
25              b[i] = (byte) (checkBounds(ints[i], 0, 256) & 0xFF);
26          }
27          return b;
28      }
29  
30      private static int checkBounds(int value, int lowerInclusive,
31                                     int upperExclusive)
32      {
33          if (value < lowerInclusive)
34          {
35              throw new IllegalArgumentException(String
36                  .format("Value %s is less than lower bound %s", value,
37                          lowerInclusive));
38          }
39          else if (value >= upperExclusive)
40          {
41              throw new IllegalArgumentException(String
42                  .format("Value %s is greater than or equal to upper bound %s",
43                          value, upperExclusive));
44          }
45          else
46          {
47              return value;
48          }
49      }
50  }
