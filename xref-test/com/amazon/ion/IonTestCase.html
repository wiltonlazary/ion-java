121     public static synchronized void loadSystemProperties()
122     {
123         if (! ourSystemPropertiesLoaded)
124         {
125             InputStream stream =
126                 IonTestCase.class.getResourceAsStream("/system.properties");
127 
128             if (stream != null)
129             {
130                 Properties props = new Properties();
131 
132                 try
133                 {
134                     props.load(stream);
135 
136                     Iterator entries = props.entrySet().iterator();
137                     while (entries.hasNext())
138                     {
139                         Map.Entry entry = (Map.Entry) entries.next();
140 
141                         String key = (String) entry.getKey();
142 
143                         // Command-line values override system.properties
144                         if (System.getProperty(key) == null)
145                         {
146                             System.setProperty(key, (String) entry.getValue());
147                         }
148                     }
149                 }
150                 catch (IOException e)
151                 {
152                     synchronized (System.out)
153                     {
154                         System.out.println("Caught exception while loading system.properties:");
155                         e.printStackTrace(System.out);
156                     }
157                 }
158             }
159 
160             ourSystemPropertiesLoaded = true;
161         }
162     }

165     public static String requireSystemProperty(String prop)
166     {
167         loadSystemProperties();
168 
169         String value = System.getProperty(prop);
170         if (value == null)
171         {
172             value = System.getenv(prop);
173         }
174         if (value == null) {
175             String message = "Missing required system property: " + prop;
176             throw new IllegalStateException(message);
177         }
178         return value;
179     }

309      * For example, <code>makeEscapedCharString('n')</code> returns

342     /**
343      * Put value into a datagram, get bytes, and reload to single value.
344      * Result should be equivalent.
345      */
346     public IonValue reload(IonValue value)
347     {
348         IonDatagram dg = system().newDatagram(value);
349         byte[] bytes = dg.getBytes();
350         checkBinaryHeader(bytes);
351 
352         try {
353             dg = loader().load(bytes);
354         } catch (final IonException e) {
355             final String hex = BinaryTest.bytesToHex(bytes);
356             throw new IonException("Bad bytes: " + hex, e);
357         }
358         assertEquals(1, dg.size());
359         return dg.get(0);
360     }

409     /**
410      * Parses text as a single Ion value.  If the text contains more than that,
411      * a failure is thrown.
412      *
413      * @param iterator must not be <code>null</code>.
414      * @return a single value, or <code>null</code> if the scanner has nothing
415      * on it.
416      */
417     public IonValue singleValue(Iterator<IonValue> iterator)
418     {
419         IonValue value = null;
420 
421         if (iterator.hasNext())
422         {
423             value = iterator.next();
424 
425             if (iterator.hasNext())
426             {
427                 IonValue part = iterator.next();
428                 fail("Found unexpected part: " + part);
429             }
430         }
431 
432         return value;
433     }

437     /**
438      * Parses text as a single Ion value.  If the text contains more than that,
439      * a failure is thrown.
440      *
441      * @param text must not be <code>null</code>.
442      * @return a single value, not <code>null</code>.
443      */
444     public IonValue oneValue(String text)
445     {
446         IonValue value = null;
447 
448         Iterator<IonValue> iterator = system().iterate(text);
449         if (iterator.hasNext())
450         {
451             value = iterator.next();
452 
453             if (iterator.hasNext())
454             {
455                 IonValue part = iterator.next();
456                 fail("Found unexpected part <" + part + "> in text: " + text);
457             }
458         }
459         else
460         {
461             fail("No data found in text: " + text);
462         }
463 
464         return value;
465     }

538     {
539         if (datagram.length != 0) // Allow empty "binary" stream
540         {
541             assertTrue("datagram is too small", datagram.length >= 4);
542 
543             assertEquals("datagram cookie byte 1", 0xE0, datagram[0] & 0xff );
544             assertEquals("datagram cookie byte 2", 0x01, datagram[1] & 0xff);
545             assertEquals("datagram cookie byte 3", 0x00, datagram[2] & 0xff);
546             assertEquals("datagram cookie byte 4", 0xEA, datagram[3] & 0xff);
547         }

823     public static void checkSymbol(String text, int sid, boolean dupe,
824                                    SymbolTable symtab)
825     {
826         assert !dupe || text != null;
827 
828         String msg = "text:" + text + " sid:" + sid;
829 
830         if (text != null)
831         {
832             if (sid != UNKNOWN_SYMBOL_ID)
833             {
834                 assertEquals(msg, text, symtab.findKnownSymbol(sid));
835             }
836 
837             // Can't do this stuff when we have a duplicate symbol.
838             if (! dupe)
839             {
840                 assertEquals(msg, sid, symtab.findSymbol(text));
841 
842                 SymbolToken symToken = symtab.intern(text);
843                 assertEquals(msg, sid, symToken.getSid());
844 
845                 symToken = symtab.find(text);
846                 assertEquals(msg, sid, symToken.getSid());
847             }
848         }
849         else // No text expected, must have sid
850         {
851             assertEquals(msg, text /* null */, symtab.findKnownSymbol(sid));
852         }
853     }

955     public static SymbolTable findImportedTable(SymbolTable localTable,
956                                                 String importName)
957     {
958         SymbolTable[] imports = localTable.getImportedTables();
959         if (imports == null) return null;
960 
961         for (int i = 0; i < imports.length; i++)
962         {
963             SymbolTable current = imports[i];
964             // FIXME why does this allow null?
965             if (current != null && importName.equals(current.getName()))
966             {
967                 return current;
968             }
969         }
970 
971         return null;
972     }
