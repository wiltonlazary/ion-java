16  package com.amazon.ion.impl.lite;
17  
18  import com.amazon.ion.IonContainer;
19  import com.amazon.ion.SymbolTable;
20  
21  /**
22   * Provides the parent, system and symbol table definitions that are shared
23   * by one or more hierarchies of IonValues.
24   * <p>
25   * The type of context used by a value depends on its position.
26   * <ul>
27   *   <li>Top-level values have either a {@link TopLevelContext} or an
28   *   {@link IonSystemLite} for a context.
29   *   <li>Non-top-level values have their {@link IonContainer} for a context.
30   *   <li>Datagrams have their {@link IonSystemLite} for a context.
31   * </ul>
32   * <p>
33   *  Concrete
34   *  contexts simply return the correct values (and
35   *  own them).  Containers return themselves as the
36   *  parent and delegate to their parent for the
37   *  system and symbol table values.  Note that values
38   *  contained directly by a datagram may store a
39   *  symbol table locally as there may be more than
40   *  one symbol table in a datagram.
41   */
42  interface IonContext
43  {
44      /**
45       * Return the container of values associated with this context.
46       * If this context is an {@link IonContainerLite} it returns itself.
47       *
48       * @return the container of the value;
49       *  null for stand-alone top level values.
50       */
51      abstract IonContainerLite getContextContainer();
52  
53      /**
54       * Get the IonSystem concrete object that created
55       * the IonValue that is associated with this context.
56       * Generally this delegates to the parent.
57       *
58       * @return not null
59       */
60      abstract IonSystemLite getSystem();
61  
62  
63      /**
64       * Returns the symbol table that is directly assigned to this context.
65       * For {@link TopLevelContext} it is the symbol table member.
66       * For {@link IonContainerLite} and {@link IonSystemLite} it is null.
67       *
68       * @return the directly assigned symbol table, with no recursive lookup.
69       */
70      abstract SymbolTable getContextSymbolTable();
71  }
